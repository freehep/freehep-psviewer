312 	private Collection<DSCEventListener> listeners = new ArrayList<DSCEventListener>();
313 	private Map<String, Object> parseTable = new HashMap<String, Object>();
314 	private Map<String, Object> dscTable = new HashMap<String, Object>();

316 	public DSC() {
317 		// fill the parseTable
318 		for (int i = 0; i < KEY_ARGS.length; i += 2) {
319 			String token = (String) KEY_ARGS[i];
320 			try {
321 				Object obj = KEY_ARGS[i + 1];
322 				if (obj instanceof Class<?>) {
323 					obj = ((Class<?>) obj).newInstance();
324 				} else if (obj instanceof String[]) {
325 					obj = new Enumeration((String[]) obj);
326 				}
327 				addDSCComment(token, obj);
328 			} catch (InstantiationException ie) {
329 				log.log(Level.WARNING, "DSC: could not instantiate class: '"
330 						+ token, ie);
331 			} catch (IllegalAccessException ie) {
332 				log.log(Level.WARNING, "DSC: no access to class: '"
333 								+ token, ie);
334 			}
335 		}
336 	}

352 	// Parses a DCS comment (not pre-pended with the %% signs, but including the
353 	// appended colon and arguments).
354 	public final boolean parse(String line, OperandStack os) {
355 		String[] tokens = line.split("[ \t]", 2);
356 		String key = tokens[0];
357 		String params = "";
358 		if (tokens.length == 1) {
359 			// special case if space is missing after colon
360 			tokens = line.split(":", 2);
361 			if (tokens.length == 2) {
362 				key = tokens[0] + ":";
363 				params = tokens[1];
364 			}
365 		} else {
366 			params = tokens[1];
367 		}
368 
369 		// lookup comment, parse arguments and derive the state
370 		Arguments args = (Arguments) parseTable.get(key);
371 		State state;
372 		if (params.startsWith("(atend)")) {
373 			dscTable.put(key, new Boolean(true));
374 			state = (args != null) ? State.PARSED : State.UNPARSED;
375 		} else {
376 			if (args != null) {
377 				Object parsed = args.parse(key, params, os);
378 				if (parsed != null) {
379 					dscTable.put(key, parsed);
380 					state = State.PARSED;
381 				} else {
382 					state = State.ERROR;
383 				}
384 			} else {
385 				dscTable.put(key, params);
386 				state = State.UNPARSED;
387 			}
388 		}
389 
390 		// inform the listeners
391 		for (Iterator<DSCEventListener> i = listeners.iterator(); i.hasNext();) {
392 			DSCEventListener listener = i.next();
393 			listener.dscCommentFound(new DSCEvent(this, key, dscTable.get(key),
394 					state));
395 		}
396 
397 		return state != State.ERROR;
398 	}

427 	public static class Rectangle implements Arguments {
428 		private Logger log = Logger.getLogger("org.freehep.postscript");
429 		
430 		public final Object parse(String key, String params, OperandStack os) {
431 			try {
432 				String[] tokens = params.trim().split("[ \t]", 4);
433 				if (tokens.length != 4) {
434 					return null;
435 				}
436 				int[] p = new int[4];
437 				for (int i = 0; i < tokens.length; i++) {
438 					p[i] = Integer.parseInt(tokens[i]);
439 				}
440 				return new java.awt.Rectangle(p[0], p[1], p[2] - p[0], p[3]
441 						- p[1]);
442 			} catch (NumberFormatException e) {
443 				log.logp(Level.WARNING, getClass().getName(), "parse", e.getMessage(), e);
444 				return null;
445 			}
446 		}
447 	}

461 		private String[] enumeration;
462 
463 		public Enumeration(String[] enumeration) {
464 			this.enumeration = new String[enumeration.length];
465 			System.arraycopy(enumeration, 0, this.enumeration, 0, enumeration.length);
466 		}
467 
468 		public final Object parse(String key, String params, OperandStack os) {
469 			String[] tokens = params.split("[ \t]", 2);
470 			for (int i = 0; i < enumeration.length; i++) {
471 				if (enumeration[i].equals(tokens[0])) {
472 					return tokens[0];
473 				}
474 			}
475 			return null;
476 		}

509 			try {

511 				if (tokens.length != 2) {

514

522 		}
