14  public abstract class AbstractOperator extends PSOperator {
15  	private static Logger slog = Logger.getLogger("org.freehep.postscript");
16  
17  	public static void register(PSDictionary dict, Class<?>[] clazz) {
18  		for (int j = 0; j < clazz.length; j++) {
19  			try {
20  				PSOperator op = (PSOperator) clazz[j].newInstance();
21  				PSName key = new PSName(op.getName());
22  				if (dict.get(key) == null) {
23  					dict.put(key, op);
24  					op.setName(key.cvs());
25  				} else {
26  					slog.severe("Duplicate operator '" + key + "'");
27  					System.exit(1);
28  				}
29  
30  			} catch (ClassCastException e) {
31  				slog.log(Level.WARNING, "Error: " + clazz
32  						+ " does not inherit from PSOperator.\n", e);
33  			} catch (IllegalAccessException e) {
34  				slog.log(Level.WARNING, "Error: " + clazz
35  						+ " cannot be instantiated.\n", e);
36  			} catch (InstantiationException e) {
37  				slog.log(Level.WARNING, "Error: " + clazz
38  						+ " cannot be instantiated.\n", e);
39  			}
40  		}
41  	}
42  }
