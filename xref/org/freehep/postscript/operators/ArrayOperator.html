17  public abstract class ArrayOperator extends AbstractOperator {
18  
19  	public static void register(PSDictionary dict) {
20  		AbstractOperator.register(dict, new Class<?>[] { Array.class,
21  				ArrayBegin.class, ArrayEnd.class, AStore.class });
22  	}
23  }

25  class Array extends ArrayOperator {
26  	{
27  		operandTypes = new Class[] { PSInteger.class };
28  	}
29  
30  	@Override
31  	public boolean execute(OperandStack os) {
32  		PSInteger n = os.popInteger();
33  		if (n.getValue() < 0) {
34  			error(os, new RangeCheck());
35  		} else {
36  			os.push(new PSArray(n.getValue()));
37  		}
38  		return true;
39  	}
40  }

56  class ArrayEnd extends ArrayOperator {
57  
58  	@Override
59  	public String getName() {
60  		return ("]");
61  	}
62  
63  	// FREEHEP-139: nothing done about InvalidAccess
64  	@Override
65  	public boolean execute(OperandStack os) {
66  		int n = os.countToMark();
67  		if (n < 0) {
68  			error(os, new UnmatchedMark());
69  		} else {
70  			PSObject[] a = new PSObject[n];
71  			for (int i = n - 1; i >= 0; i--) {
72  				PSObject o = os.popObject();
73  				a[i] = o;
74  			}
75  			/* PSMark mark = */os.popMark();
76  
77  			os.push(new PSArray(a));
78  		}
79  		return true;
80  	}
81  }

83  class AStore extends ArrayOperator {
84  	{
85  		operandTypes = new Class[] { PSArray.class };
86  	}
87  
88  	@Override
89  	public boolean execute(OperandStack os) {
90  		PSArray a = os.popArray();
91  		int n = a.size();
92  		if (n > os.size()) {
93  			error(os, new StackUnderflow());
94  		} else {
95  			for (int i = n - 1; i >= 0; i--) {
96  				PSObject o = os.popObject();
97  				a.set(i, o);
98  			}
99  			os.push(a);
100 		}
101 		return true;
102 	}
