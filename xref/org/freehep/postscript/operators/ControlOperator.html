53  	}
54  
55  	@Override
56  	public boolean execute(OperandStack os) {
57  		PSPackedArray p = os.popPackedArray();
58  		PSBoolean b = os.popBoolean();
59  		if (b.getValue()) {
60  			os.execStack().pop();
61  			os.execStack().push(p);
62  			return false;
63  		}
64  		return true;
65  	}

68  class IfElse extends ControlOperator {
69  	{
70  		operandTypes = new Class[] { PSBoolean.class, PSPackedArray.class,
71  				PSPackedArray.class };
72  	}
73  
74  	@Override
75  	public boolean execute(OperandStack os) {
76  		PSPackedArray p2 = os.popPackedArray();
77  		PSPackedArray p1 = os.popPackedArray();
78  		PSBoolean b = os.popBoolean();
79  		os.execStack().pop();
80  		if (b.getValue()) {
81  			os.execStack().push(p1);
82  		} else {
83  			os.execStack().push(p2);
84  		}
85  		return false;
86  	}

89  class For extends ControlOperator implements LoopingContext {
90  
91  	private PSPackedArray procedure = null;
92  	private int iLimit = 0;
93  	private int iIncrement = 0;
94  	private int iCount = 0;
95  	private double rLimit = 0.0;
96  	private double rIncrement = 0.0;
97  	private double rCount = 0.0;
98  	private boolean real = false;
99  
100 	public For() {
101 	}
102 
103 	private For(int start, int inc, int limit, PSPackedArray p) {
104 		iCount = start;
105 		iIncrement = inc;
106 		iLimit = limit;
107 		procedure = p;
108 		real = false;
109 	}
110 
111 	public For(double start, double inc, double limit, PSPackedArray p) {
112 		rCount = start;
113 		rIncrement = inc;
114 		rLimit = limit;
115 		procedure = p;
116 		real = true;
117 	}
118 
119 	@Override
120 	public boolean execute(OperandStack os) {
121 
122 		if (procedure == null) {
123 			if (!os.checkType(PSNumber.class, PSNumber.class, PSNumber.class,
124 					PSPackedArray.class)) {
125 				error(os, new TypeCheck());
126 				return true;
127 			}
128 
129 			PSPackedArray p = os.popPackedArray();
130 			PSNumber limit = os.popNumber();
131 			PSNumber inc = os.popNumber();
132 			PSNumber start = os.popNumber();
133 			os.execStack().pop();
134 			if ((limit instanceof PSReal) || (inc instanceof PSReal)
135 					|| (start instanceof PSReal)) {
136 				os.execStack().push(
137 						new For(start.getDouble(), inc.getDouble(), limit
138 								.getDouble(), p));
139 			} else {
140 				os.execStack().push(
141 						new For(((PSInteger) start).getValue(),
142 								((PSInteger) inc).getValue(),
143 								((PSInteger) limit).getValue(), p));
144 			}
145 			return false;
146 		}
147 
148 		if (real) {
149 			if ((rIncrement >= 0.0) && (rCount <= rLimit)) {
150 				os.push(rCount);
151 				os.execStack().push(procedure);
152 				rCount += rIncrement;
153 				return false;
154 			} else if ((rIncrement < 0.0) && (rCount >= rLimit)) {
155 				os.push(rCount);
156 				os.execStack().push(procedure);
157 				rCount += rIncrement;
158 				return false;
159 			}
160 		} else {
161 			if ((iIncrement >= 0) && (iCount <= iLimit)) {
162 				os.push(iCount);
163 				os.execStack().push(procedure);
164 				iCount += iIncrement;
165 				return false;
166 			} else if ((iIncrement < 0) && (iCount >= iLimit)) {
167 				os.push(iCount);
168 				os.execStack().push(procedure);
169 				iCount += iIncrement;
170 				return false;
171 			}
172 		}
173 		return true;
174 	}
175 }

177 class Repeat extends ControlOperator implements LoopingContext {
178 
179 	private PSPackedArray procedure = null;
180 	private int limit = 0;
181 	private int i = 0;
182 
183 	public Repeat() {
184 	}
185 
186 	private Repeat(int n, PSPackedArray p) {
187 		procedure = p;
188 		limit = n;
189 		i = 0;
190 	}
191 
192 	@Override
193 	public boolean execute(OperandStack os) {
194 
195 		if (procedure == null) {
196 			if (!os.checkType(PSInteger.class, PSPackedArray.class)) {
197 				error(os, new TypeCheck());
198 				return true;
199 			}
200 
201 			PSPackedArray p = os.popPackedArray();
202 			PSInteger n = os.popInteger();
203 			if (n.getValue() < 0) {
204 				error(os, new RangeCheck());
205 				return true;
206 			}
207 			os.execStack().pop();
208 			os.execStack().push(new Repeat(n.getValue(), p));
209 			return false;
210 		}
211 
212 		if (i < limit) {
213 			os.execStack().push(procedure);
214 			i++;
215 			return false;
216 		}
217 		return true;
218 	}

250 class Exit extends ControlOperator {
251 
252 	@Override
253 	public boolean execute(OperandStack os) {
254 		PSObject o = os.execStack().popObject();
255 		while ((o != null) && !(o instanceof LoopingContext)) {
256 			o = os.execStack().popObject();
257 		}
258 		if (o == null) {
259 			log.severe("Error: No enclosing LoopingContext; Command: exit");
260 			os.execStack().push("quit");
261 			os.execStack().push(
262 					os.dictStack().errorDictionary().get("handleerror"));
263 			return false;
264 		}
265 
266 		if ((o instanceof Stopped) || (o instanceof Run)) {
267 			os.execStack().push(o);
268 			os.execStack().push("invalidexit");
269 			return false;
270 		}
271 
272 		// ok
273 		return false;
274 	}

277 class Stop extends ControlOperator {
278 
279 	// Our stop operator will print the error message if not in a stopped
280 	// context
281 	@Override
282 	public boolean execute(OperandStack os) {
283 
284 		PSObject o = os.execStack().popObject();
285 		while ((o != null) && !(o instanceof Stopped)) {
286 			o = os.execStack().popObject();
287 		}
288 		if (o == null) {
289 			// FIXME: this should be part of a standard stopped context
290 			// just a message written here
291 			os.execStack().push("quit");
292 			os.execStack().push(
293 					os.dictStack().errorDictionary().get("handleerror"));
294 		} else {
295 			// do as if stopped returned true
296 			os.push(true);
297 		}
298 		return false;
299 	}

302 class Stopped extends ControlOperator implements LoopingContext {
303 
304 	private boolean stopped = true;
305 
306 	public Stopped() {
307 	}
308 
309 	private Stopped(boolean state) {
310 		stopped = state;
311 	}
312 
313 	@Override
314 	public boolean execute(OperandStack os) {
315 		if (stopped) {
316 			if (!os.checkType(PSObject.class)) {
317 				error(os, new TypeCheck());
318 				return true;
319 			}
320 			PSObject obj = os.popObject();
321 
322 			os.execStack().pop();
323 			os.execStack().push(new Stopped(false));
324 			os.execStack().push(obj);
325 			return false;
326 		}
327 
328 		os.push(false);
329 		return true;
330 	}

376 class EExec extends ControlOperator {
377 	{
378 		operandTypes = new Class[] { PSDataSource.class };
379 	}
380 
381 	@Override
382 	public boolean execute(OperandStack os) {
383 
384 		if (os.checkType(PSDataSource.class)) {
385 			try {
386 				PSDataSource source = os.popDataSource();
387 				os.execStack().pop();
388 				os.execStack().push(
389 						new PSInputFile(new EEXECDecryption(source
390 								.getInputStream(), EEXECConstants.EEXEC_R,
391 								EEXECConstants.N), source.getDSC()));
392 			} catch (IOException e) {
393 				error(os, new InvalidFileAccess());
394 			}
395 			return false;
396 		} else {
397 			error(os, new TypeCheck());
398 			return true;
399 		}
400 	}
