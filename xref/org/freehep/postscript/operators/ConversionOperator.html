156 class CvI extends ConversionOperator {
157 	{
158 		operandTypes = new Class[] { PSObject.class };
159 	}
160 
161 	@Override
162 	public boolean execute(OperandStack os) {
163 		try {
164 			if (os.checkType(PSNumber.class)) {
165 				PSNumber n = os.popNumber();
166 				os.push(n.getInt());
167 			} else if (os.checkType(PSString.class)) {
168 				PSString s = os.popString();
169 				PSNumber n = PSUtils.parseNumber(s.getValue());
170 				os.push(n.getInt());
171 			} else {
172 				error(os, new TypeCheck());
173 			}
174 		} catch (NumberFormatException e) {
175 			error(os, new UndefinedResult());
176 		} catch (RangeException e) {
177 			error(os, new RangeCheck());
178 		}
179 
180 		return true;
181 	}
182 }

222 class CvRS extends ConversionOperator {
223 	{
224 		operandTypes = new Class[] { PSNumber.class, PSInteger.class,
225 				PSString.class };
226 	}
227 
228 	@Override
229 	public boolean execute(OperandStack os) {
230 		try {
231 			PSString s = os.popString();
232 			PSInteger r = os.popInteger();
233 			PSNumber n = os.popNumber();
234 			byte[] b;
235 			if (r.getValue() == 10) {
236 				b = n.cvs().getBytes();
237 			} else {
238 				b = Integer.toString(n.getInt(), r.getValue()).getBytes();
239 			}
240 			if (b.length > s.size()) {
241 				error(os, new RangeCheck());
242 			} else {
243 				for (int i = 0; i < b.length; i++) {
244 					s.set(i, b[i]);
245 				}
246 				os.push(s.subString(0, b.length));
247 			}
248 		} catch (RangeException e) {
249 			error(os, new RangeCheck());
250 		}
251 
252 		return true;
253 	}
