19  /**
20   * Error Operators for PostScript Processor
21   * 
22   * @author Mark Donszelmann
23   */
24  public abstract class ErrorOperator extends AbstractOperator {
25  	protected Logger log = Logger.getLogger("org.freehep.postscript");
26  
27  	public static void register(PSDictionary dict) {
28  		AbstractOperator.register(dict, new Class<?>[] {
29  				ConfigurationError.class, DictFull.class,
30  				DictStackOverflow.class, DictStackUnderflow.class,
31  				ExecStackOverflow.class, HandleError.class, Interrupt.class,
32  				InvalidAccess.class, InvalidExit.class,
33  				InvalidFileAccess.class, InvalidFont.class,
34  				InvalidRestore.class, IOError.class, LimitCheck.class,
35  				NoCurrentPoint.class, RangeCheck.class, StackOverflow.class,
36  				StackUnderflow.class, SyntaxError.class, Timeout.class,
37  				TypeCheck.class, Undefined.class, UndefinedFileName.class,
38  				UndefinedResource.class, UndefinedResult.class,
39  				UnmatchedMark.class, Unregistered.class, VMError.class,
40  				Unimplemented.class });
41  	}
42  
43  	// default error handler
44  	@Override
45  	public boolean execute(OperandStack os) {
46  		// fill error dict
47  		PSDictionary error = os.dictStack().dollarError();
48  		error.put("newerror", new PSBoolean(true));
49  		error.put("errorname", new PSName(getName()));
50  		error.put("command", os.execStack().peekObject(1));
51  		if (error.get("recordstacks").equals(new PSBoolean(true))) {
52  			error.put("ostack", stackToArray(os));
53  			error.put("estack", stackToArray(os.execStack()));
54  			error.put("dstack", stackToArray(os.dictStack()));
55  		}
56  
57  		// now execute stop
58  		os.execStack().pop();
59  		os.execStack().push("stop");
60  
61  		return false;
62  	}
63  
64  	private static PSArray stackToArray(PostScriptStack stack) {
65  		PSArray array = new PSArray(stack.size());
66  		stack.copyInto(array);
67  		return array;
68  	}
69  }

86  class HandleError extends ErrorOperator {
87  
88  	// FIXME, should print more
89  	@Override
90  	public boolean execute(OperandStack os) {
91  		// report on error
92  		PSDictionary error = os.dictStack().dollarError();
93  		if (error.get("newerror").equals(new PSBoolean(true))) {
94  			log.warning("\n\n%%[Error: " + error.get("errorname").toPrint()
95  					+ "; Offending Command: " + error.get("command").toPrint()
96  					+ "]%%");
97  			log.warning("");
98  
99  			if (!error.get("errorinfo").equals(new PSNull())) {
100 				log.warning("Error Info: " + error.get("errorinfo").toPrint());
101 				log.warning("");
102 			}
103 
104 			if (error.get("recordstacks").equals(new PSBoolean(true))) {
105 				log.warning("Operand Stack (bottom..top)");
106 				log.warning(error.get("ostack").toPrint());
107 				log.warning("");
108 
109 				log.warning("Execution Stack (bottom..top)");
110 				log.warning(error.get("estack").toPrint());
111 				log.warning("");
112 
113 				log.warning("Dictionary Stack (bottom..top)");
114 				log.warning(error.get("dstack").toPrint());
115 				log.warning("");
116 			}
117 
118 			// reset error dict
119 			error.put("newerror", new PSBoolean(false));
120 			error.put("errorinfo", new PSNull());
121 		}
122 
123 		return true;
124 	}
