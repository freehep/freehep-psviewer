62  class FileFile extends FileOperator {
63  	{
64  		operandTypes = new Class[] { PSString.class, PSString.class };
65  	}
66  
67  	@Override
68  	public String getName() {
69  		return "file";
70  	}
71  
72  	@Override
73  	public boolean execute(OperandStack os) {
74  		String access = os.popString().getValue();
75  		String filename = os.popString().getValue();
76  		try {
77  			if (filename.equals("%stdin")) {
78  				os.push(new PSInputFile(System.in, os.getDSC()));
79  
80  			} else if (filename.equals("%stdout")) {
81  				os.push(new PSOutputFile(System.out));
82  
83  			} else if (filename.equals("%stderr")) {
84  				os.push(new PSOutputFile(System.err));
85  
86  			} else if (filename.equals("%lineedit")
87  					|| filename.equals("%statementedit")) {
88  				error(os, new Undefined(
89  						"%lineedit and %statementedit not supported."));
90  
91  			} else {
92  				// regular file
93  				boolean rw = (access.length() > 1) && (access.charAt(1) == '+');
94  				boolean write = (access.length() > 0)
95  						&& (access.charAt(0) == 'w');
96  				boolean append = (access.length() > 0)
97  						&& (access.charAt(0) == 'a');
98  
99  				if (rw) {
100 					os.push(new PSRandomAccessFile(filename, write, append, os
101 							.isSecure()));
102 				} else {
103 					if (write || append) {
104 						os.push(new PSOutputFile(filename, append, os
105 								.isSecure()));
106 					} else {
107 						os.push(new PSInputFile(filename, os.getDSC()));
108 					}
109 				}
110 			}
111 		} catch (FileNotFoundException e) {
112 			error(os, new UndefinedFileName());
113 		} catch (IOException e) {
114 			error(os, new InvalidFileAccess());
115 		}
116 		return true;
117 	}

121 	{
122 		operandTypes = new Class[] { PSName.class };

124 
125 	public boolean encode(OperandStack os, String filterName)
126 			throws IOException {
127 		if (filterName.equals("RunLengthEncode")) {
128 			// special parameter
129 			/* int runLength = */os.popInteger().getValue();
130 			// PSDictionary dict;
131 			if (os.checkType(PSDictionary.class)) {
132 				/* dict = */os.popDictionary();
133 			}
134 			PSDataTarget target = os.popDataTarget();
135 			// FIXME: runLength parameter is ignored
136 			os.push(new PSOutputFile(new RunLengthOutputStream(target
137 					.getOutputStream())));
138 
139 			error(os, new Undefined());
140 			return true;
141 
142 		} else {
143 			// no parameters, optional dictionary
144 			// PSDictionary dict;
145 			if (os.checkType(PSDictionary.class)) {
146 				/* dict = */os.popDictionary();
147 			}
148 			PSDataTarget target = os.popDataTarget();
149 			if (filterName.equals("ASCIIHexEncode")) {
150 				os.push(new PSOutputFile(new ASCIIHexOutputStream(target
151 						.getOutputStream())));
152 
153 			} else if (filterName.equals("ASCII85Encode")) {
154 				os.push(new PSOutputFile(new ASCII85OutputStream(target
155 						.getOutputStream())));
156 
157 			} else if (filterName.equals("LZWEncode")) {
158 				error(os, new Undefined());
159 				return true;
160 
161 			} else if (filterName.equals("FlateEncode")) {
162 				os.push(new PSOutputFile(new FlateOutputStream(target
163 						.getOutputStream())));
164 
165 			} else if (filterName.equals("CCITTFaxEncode")) {
166 				error(os, new Undefined());
167 				return true;
168 
169 			} else if (filterName.equals("NullEncode")) {
170 				os.push(new PSOutputFile(target.getOutputStream()));
171 
172 			} else {
173 				error(os, new Undefined());
174 				return true;
175 			}
176 		}
177 
178 		return true;
179 	}
180 
181 	public boolean decode(OperandStack os, String filterName)
182 			throws IOException {
183 		if (filterName.equals("SubFileDecode")) {
184 			// special parameters
185 			error(os, new Undefined());
186 			return true;
187 		} else {
188 			// no parameters, optional dictionary
189 			// PSDictionary dict;
190 			if (os.checkType(PSDictionary.class)) {
191 				/* dict = */os.popDictionary();
192 			}
193 			PSDataSource source = os.popDataSource();
194 			if (filterName.equals("ASCIIHexDecode")) {
195 				os.push(new PSInputFile(new ASCIIHexInputStream(source
196 						.getInputStream()), source.getDSC()));
197 
198 			} else if (filterName.equals("ASCII85Decode")) {
199 				os.push(new PSInputFile(new ASCII85InputStream(source
200 						.getInputStream()), source.getDSC()));
201 
202 			} else if (filterName.equals("LZWDecode")) {
203 				error(os, new Undefined());
204 				return true;
205 
206 			} else if (filterName.equals("FlateDecode")) {
207 				os.push(new PSInputFile(new FlateInputStream(source
208 						.getInputStream()), source.getDSC()));
209 
210 			} else if (filterName.equals("RunLengthDecode")) {
211 				os.push(new PSInputFile(new RunLengthInputStream(source
212 						.getInputStream()), source.getDSC()));
213 
214 			} else if (filterName.equals("CCITTFaxDecode")) {
215 				error(os, new Undefined());
216 				return true;
217 
218 			} else if (filterName.equals("DCTDecode")) {
219 				os.push(new PSInputFile(new DCTInputStream(source
220 						.getInputStream()), source.getDSC()));
221 				return true;
222 
223 			} else if (filterName.equals("ReusableStreamDecode")) {
224 				// Level 3
225 				error(os, new Undefined());
226 				return true;
227 
228 			} else {
229 				error(os, new Undefined());
230 				return true;
231 			}
232 		}
233 
234 		if (!os.checkType(PSDataSource.class)) {
235 			error(os, new TypeCheck());
236 			return true;
237 		}
238 
239 		return true;
240 	}
241 
242 	@Override
243 	public boolean execute(OperandStack os) {
244 		String filterName = os.popName().getValue();
245 		try {
246 			return filterName.endsWith("Encode") ? encode(os, filterName)
247 					: decode(os, filterName);
248 		} catch (ClassCastException e) {
249 			error(os, new TypeCheck());
250 			return true;
251 		} catch (IOException e) {
252 			error(os, new IOError(e));
253 			return true;
254 		}
255 	}

317 class ReadHexString extends FileOperator {
318 	{
319 		operandTypes = new Class[] { PSFile.class, PSString.class };
320 	}
321 
322 	@Override
323 	public boolean execute(OperandStack os) {
324 		PSString string = os.popString();
325 		PSFile file = os.popFile();
326 		if (string.size() <= 0) {
327 			error(os, new RangeCheck());
328 		} else {
329 			try {
330 				InputStream in = new ASCIIHexInputStream(((PSInputFile) file)
331 						.getInputStream(), true);
332 				for (int i = 0; i < string.size(); i++) {
333 					int b = in.read();
334 					if (b < 0) {
335 						os.push(string.subString(0, i));
336 						os.push(false);
337 						return true;
338 					} else {
339 						string.set(i, b & 0xFF);
340 					}
341 				}
342 				os.push(string.subString(0));
343 				os.push(true);
344 			} catch (IOException e) {
345 				error(os, new IOError(e));
346 			} catch (ClassCastException e) {
347 				error(os, new InvalidAccess());
348 			}
349 		}
350 		return true;
351 	}

440 	public boolean execute(OperandStack os) {

442 		PSFile file = os.popFile();
443 		try {
444 			String line = file.readLine();
445 			if (line == null) {
446 				os.push(string.subString(0, 0));
447 				os.push(false);
448 				return true;
449 			}
450 
451 			if (string.size() < line.length()) {
452 				error(os, new RangeCheck());
453 			} else {
454 				for (int i = 0; i < line.length(); i++) {
455 					string.set(i, line.charAt(i));
456 				}
457 				os.push(string.subString(0, line.length()));
458 				os.push(true);
459 			}
460 		} catch (IOException e) {
461 			error(os, new IOError(e));
462 		}
463 		return true;
464 	}

530 class Status extends FileOperator {
531 	{
532 		operandTypes = new Class[] { PSObject.class };
533 	}
534 
535 	@Override
536 	public boolean execute(OperandStack os) {
537 		if (os.checkType(PSFile.class)) {
538 			PSFile file = os.popFile();
539 			os.push(file.isValid());
540 		} else if (os.checkType(PSString.class)) {
541 			PSString name = os.popString();
542 			File file = new File(name.getValue());
543 			if (file.exists() && file.isFile()) {
544 				os.push(file.length() / 1024); // pages
545 				os.push(file.length()); // bytes
546 				os.push(file.lastModified()); // referenced
547 				os.push(file.lastModified()); // created
548 				os.push(true);
549 			} else {
550 				os.push(false);
551 			}
552 		} else {
553 			error(os, new TypeCheck());
554 		}
555 		return true;
556 	}

618 class RenameFile extends FileOperator {
619 	{
620 		operandTypes = new Class[] { PSString.class, PSString.class };
621 	}
622 
623 	@Override
624 	public boolean execute(OperandStack os) {
625 		if (!os.isSecure()) {
626 			error(os, new InvalidFileAccess());
627 			return true;
628 		}
629 
630 		PSString name2 = os.popString();
631 		PSString name1 = os.popString();
632 		if ((name1.get(0) == '%') || (name2.get(0) == '%')) {
633 			error(os, new UndefinedFileName());
634 		} else {
635 			File file1 = new File(name1.getValue());
636 			File file2 = new File(name2.getValue());
637 			if (!file1.exists()) {
638 				error(os, new UndefinedFileName());
639 			} else {
640 				if (!file1.renameTo(file2)) {
641 					error(os, new IOError(new IOException("Could not rename "
642 							+ file1 + " to " + file2)));
643 				}
644 			}
645 		}
646 		return true;
647 	}

650 class FilenameForAll extends FileOperator implements LoopingContext {
651 
652 	private int index;
653 	private String[] files;
654 	private PSPackedArray proc;
655 	private PSString scratch;
656 
657 	public FilenameForAll() {
658 	}
659 
660 	private FilenameForAll(String[] f, PSPackedArray p, PSString s) {
661 		files = new String[f.length];
662 		System.arraycopy(f, 0, files, 0, f.length);
663 		proc = p;
664 		scratch = s;
665 		index = 0;
666 	}
667 
668 	// FREEHEP-128: incomplete
669 	@Override
670 	public boolean execute(OperandStack os) {
671 		if (proc == null) {
672 			if (!os.checkType(PSString.class, PSPackedArray.class,
673 					PSString.class)) {
674 				error(os, new TypeCheck());
675 				return true;
676 			}
677 
678 			PSString s = os.popString();
679 			PSPackedArray p = os.popPackedArray();
680 			PSString template = os.popString();
681 
682 			String dir = template.getValue();
683 			if (dir.startsWith("%")) {
684 				error(os,
685 						new Undefined("%device%file currently not supported."));
686 				return true;
687 			}
688 
689 			File directory = new File((dir.lastIndexOf('/') < 0) ? "./" : dir);
690 			FileFilter filter = new StandardFileFilter(dir);
691 			File[] matchedFiles = directory.listFiles(filter);
692 
693 			Collection<String> matchedFileNames = new ArrayList<String>();
694 			for (int i = 0; i < matchedFiles.length; i++) {
695 				matchedFileNames.add(matchedFiles[i].getName());
696 			}
697 
698 			String[] f = matchedFileNames.toArray(new String[matchedFileNames
699 					.size()]);
700 			os.execStack().pop();
701 			os.execStack().push(new FilenameForAll(f, p, s));
702 			return false;
703 		}
704 
705 		if (index >= files.length) {
706 			return true;
707 		}
708 
709 		PSString name = scratch.set(files[index]);
710 		os.push(name);
711 		os.execStack().push(proc);
712 		index++;
713 		return true;
714 	}
