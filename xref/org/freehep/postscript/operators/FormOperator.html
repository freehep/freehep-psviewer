33  class MakePattern extends FormOperator {
34  
35  	private PSDictionary pattern;
36  	private Paint paint;
37  
38  	private MakePattern(PSDictionary d, Paint p) {
39  		pattern = d;
40  		paint = p;
41  	}
42  
43  	public MakePattern() {
44  	}
45  
46  	@Override
47  	public boolean execute(OperandStack os) {
48  		if (pattern == null) {
49  			if (!os.checkType(PSDictionary.class, PSPackedArray.class)) {
50  				error(os, new TypeCheck());
51  				return true;
52  			}
53  
54  			AffineTransform m = new AffineTransform(os.popPackedArray()
55  					.toDoubles());
56  			PSDictionary d = os.popDictionary();
57  
58  			// FIXME: no checking
59  			// FIXME: no copy
60  			int type = d.getInteger("PatternType");
61  			switch (type) {
62  			case 1: // tiling pattern
63  				PSPackedArray proc = (PSPackedArray) d.getPackedArray(
64  						"PaintProc").copy();
65  				double[] bbox = d.getPackedArray("BBox").toDoubles();
66  				double xStep = d.getNumber("XStep").getDouble();
67  				double yStep = d.getNumber("YStep").getDouble();
68  				int paintType = d.getInteger("PaintType");
69  				// FIXME: ignored
70  				/* int tilingType = */
71  				d.getInteger("TilingType");
72  
73  				// gsave, create image, set modified gstate
74  				os.push(d);
75  				os.gsave();
76  
77  				AffineTransform inverse = new AffineTransform();
78  				try {
79  					inverse = m.createInverse();
80  				} catch (NoninvertibleTransformException e) {
81  					log.warning("Internal MakePattern Error");
82  				}
83  
84  				Point2D bb = inverse.deltaTransform(
85  						new Point2D.Double(1.0, 1.0), null);
86  				double w = bbox[2] - bbox[0] + bb.getX();
87  				double h = bbox[3] - bbox[1] + bb.getY();
88  
89  				Point2D biSize = m.deltaTransform(new Point2D.Double(xStep,
90  						yStep), null);
91  				int biWidth = (int) biSize.getX();
92  				int biHeight = (int) biSize.getY();
93  				BufferedImage bi = os.gstate()
94  						.convertToImage(biWidth, biHeight);
95  
96  				AffineTransform ctm = os.gstate().getTransform();
97  
98  				os.gstate().setTransform(m);
99  				// FIXME: offset may not be completely correct (HEART2)
100 				os.gstate().translate(-bbox[0], -bbox[1]);
101 
102 				os.gstate().newPath();
103 				os.gstate()
104 						.clip(new Rectangle2D.Double(bbox[0], bbox[1], w, h));
105 				Point2D offset = inverse.transform(new Point2D.Double(bbox[0],
106 						bbox[1]), null);
107 				Rectangle2D box = new Rectangle2D.Double(offset.getX(), offset
108 						.getY(), biWidth, biHeight);
109 				Paint p = new FixedTexturePaint(os, ctm, bi, box);
110 
111 				os.execStack().pop();
112 				os.execStack().push(new MakePattern(d, p));
113 				os.execStack().push(proc);
114 
115 				switch (paintType) {
116 				case 1:
117 					break;
118 				case 2:
119 					os.gstate().setColorSpace("DeviceRGB");
120 					os.gstate().setColor(new float[] { 1.0f, 0.0f, 0.0f });
121 					break;
122 				default:
123 					error(os, new RangeCheck());
124 					return true;
125 				}
126 				return false;
127 			default:
128 				// FIXME
129 				error(os, new RangeCheck());
130 				return true;
131 			}
132 		}
133 
134 		os.grestore();
135 
136 		// retrieve paint and save as implementation
137 		pattern.put("Implementation", new PSPaint(paint));
138 
139 		os.push(pattern);
140 		return true;
141 	}
142 }

144 class SetPattern extends FormOperator {
145 	{
146 		operandTypes = new Class[] { PSDictionary.class };
147 	}
148 
149 	@Override
150 	public boolean execute(OperandStack os) {
151 		PSDictionary pattern = os.popDictionary();
152 
153 		// FIXME: no checking
154 		int type = pattern.getInteger("PatternType");
155 		switch (type) {
156 		case 1: // tiling
157 			PSPaint implementation = (PSPaint) pattern.get("Implementation");
158 			Paint paint = implementation.getValue();
159 			int paintType = pattern.getInteger("PaintType");
160 			switch (paintType) {
161 			case 1:
162 				os.gstate().setColor(paint);
163 				break;
164 			case 2:
165 				int n = os.gstate().getNumberOfColorSpaceComponents();
166 				PSObject[] params = new PSObject[n];
167 				for (int i = n - 1; i >= 0; i--) {
168 					params[i] = os.popNumber();
169 				}
170 				os.gstate().setColor(paint, params);
171 				break;
172 			default:
173 				error(os, new RangeCheck());
174 				return true;
175 			}
176 			break;
177 		default:
178 			error(os, new RangeCheck());
179 			break;
180 		}
181 		return true;
182 	}
183 }

185 class ExecForm extends FormOperator {
186 	private boolean done;
187 
188 	private ExecForm(boolean d) {
189 		done = d;
190 	}
191 
192 	public ExecForm() {
193 	}
194 
195 	@Override
196 	public boolean execute(OperandStack os) {
197 		if (!done) {
198 			if (!os.checkType(PSDictionary.class)) {
199 				error(os, new TypeCheck());
200 				return true;
201 			}
202 			PSDictionary form = os.popDictionary();
203 
204 			try {
205 				if (form.getInteger("FormType") != 1) {
206 					error(os, new RangeCheck());
207 					return true;
208 				}
209 
210 				AffineTransform matrix = new AffineTransform(form
211 						.getPackedArray("Matrix").toDoubles());
212 				double[] bbox = form.getPackedArray("BBox").toDoubles();
213 				PSPackedArray proc = (PSPackedArray) form.getPackedArray(
214 						"PaintProc").clone();
215 				proc.setExecutable();
216 
217 				// gsave, concat matrix, rectclip bbox
218 				os.gsave();
219 				os.gstate().transform(matrix);
220 				os.gstate().clip(
221 						new Rectangle2D.Double(bbox[0], bbox[1], bbox[2]
222 								- bbox[0], bbox[3] - bbox[1]));
223 
224 				os.execStack().pop();
225 				os.execStack().push(new ExecForm(true));
226 				os.execStack().push(proc);
227 				os.push(form);
228 				return false;
229 			} catch (ClassCastException e) {
230 				error(os, new TypeCheck());
231 				return true;
232 			} catch (CloneNotSupportedException e) {
233 				error(os, new Unimplemented());
234 				return true;
235 			}
236 		}
237 
238 		os.grestore();
239 		return true;
240 	}
