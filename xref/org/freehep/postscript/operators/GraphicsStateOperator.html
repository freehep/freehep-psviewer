274 class SetColorSpace extends GraphicsStateOperator {
275 	{
276 		operandTypes = new Class[] { PSObject.class };
277 	}
278 
279 	@Override
280 	public boolean execute(OperandStack os) {
281 		String name;
282 		Object[] params;
283 		if (os.checkType(PSPackedArray.class)) {
284 			PSPackedArray a = os.popPackedArray();
285 			name = a.get(0).cvs();
286 			params = a.toObjects();
287 		} else if (os.checkType(PSName.class)) {
288 			name = os.popName().cvs();
289 			params = null;
290 		} else {
291 			name = "Undefined";
292 			params = null;
293 		}
294 
295 		if (!os.gstate().setColorSpace(name, params)) {
296 			error(os, new Undefined());
297 		}
298 		return true;
299 	}
300 }

318 class SetColor extends GraphicsStateOperator {
319 	{
320 		operandTypes = new Class[] { PSObject.class };
321 	}
322 
323 	// FIXME: handles only RGB, GRAY and CMYK
324 	@Override
325 	public boolean execute(OperandStack os) {
326 		if (os.checkType(PSNumber.class)) {
327 			String space = os.gstate().colorSpace();
328 			if (space.equals(Constants.DEVICE_GRAY)) {
329 				PSNumber g = os.popNumber();
330 				os.gstate().setColor(new float[] { g.getFloat() });
331 			} else if (space.equals(Constants.DEVICE_RGB)) {
332 				PSNumber blue = os.popNumber();
333 				PSNumber green = os.popNumber();
334 				PSNumber red = os.popNumber();
335 				os.gstate().setColor(
336 						new float[] { red.getFloat(), green.getFloat(),
337 								blue.getFloat() });
338 			} else if (space.equals(Constants.DEVICE_CMYK)) {
339 				PSNumber black = os.popNumber();
340 				PSNumber yellow = os.popNumber();
341 				PSNumber magenta = os.popNumber();
342 				PSNumber cyan = os.popNumber();
343 				os.gstate().setColor(
344 						new float[] { cyan.getFloat(), magenta.getFloat(),
345 								yellow.getFloat(), black.getFloat() });
346 			} else {
347 				error(os, new Undefined());
348 			}
349 		} else if (os.checkType(PSDictionary.class)) {
350 			PSDictionary pattern = os.popDictionary();
351 			PSPaint paint = (PSPaint) pattern.get("Implementation");
352 			int patternType = pattern.getInteger("PatternType");
353 			int paintType = pattern.getInteger("PaintType");
354 			if ((patternType == 1) && (paintType == 2)) {
355 				int n = os.gstate().getNumberOfColorSpaceComponents();
356 				PSObject[] params = new PSObject[n];
357 				for (int i = n - 1; i >= 0; i--) {
358 					params[i] = os.popNumber();
359 				}
360 				os.gstate().setColor(paint.getValue(), params);
361 			} else {
362 				os.gstate().setColor(paint.getValue());
363 			}
364 
365 		} else {
366 			error(os, new TypeCheck());
367 		}
368 		return true;
369 	}

499 class CurrentCMYKColor extends GraphicsStateOperator {
500 
501 	private float[] cmyk;
502 	private float bg;
503 	private float ucr;
504 
505 	public CurrentCMYKColor() {
506 	}
507 
508 	private CurrentCMYKColor(float[] color) {
509 		cmyk = new float[color.length];
510 		System.arraycopy(color, 0, cmyk, 0, color.length);
511 		bg = -1.0f;
512 		ucr = -1.0f;
513 	}
514 
515 	@Override
516 	public boolean execute(OperandStack os) {
517 		if (cmyk == null) {
518 			float[] color = os.gstate().color("DeviceCMYK");
519 
520 			os.execStack().pop();
521 			os.execStack().push(new CurrentCMYKColor(color));
522 			os.push(color[3]); // K
523 			os.execStack().push(os.gstate().blackGeneration().copy());
524 			return false;
525 		}
526 
527 		if (bg < 0) {
528 			bg = (os.popNumber()).getFloat();
529 
530 			os.push(cmyk[3]);
531 			os.execStack().push(os.gstate().underColorRemoval().copy());
532 			return false;
533 		}
534 
535 		ucr = (os.popNumber()).getFloat();
536 
537 		cmyk[0] = Math.min(1.0f, Math.max(0.0f, cmyk[0] - ucr));
538 		cmyk[1] = Math.min(1.0f, Math.max(0.0f, cmyk[1] - ucr));
539 		cmyk[2] = Math.min(1.0f, Math.max(0.0f, cmyk[2] - ucr));
540 		cmyk[3] = Math.min(1.0f, Math.max(0.0f, bg));
541 
542 		for (int i = 0; i < cmyk.length; i++) {
543 			os.push(cmyk[i]);
544 		}
545 		return true;
546 	}
