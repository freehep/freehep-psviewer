117 class Translate extends MatrixOperator {
118 	{
119 		operandTypes = new Class[] { PSObject.class };
120 	}
121 
122 	@Override
123 	public boolean execute(OperandStack os) {
124 		if (os.checkType(PSNumber.class, PSNumber.class)) {
125 			double ty = os.popNumber().getDouble();
126 			double tx = os.popNumber().getDouble();
127 			os.gstate().translate(tx, ty);
128 		} else if (os.checkType(PSNumber.class, PSNumber.class, PSArray.class)) {
129 			PSArray a = os.popArray();
130 			if (a.size() >= 6) {
131 				PSNumber ty = os.popNumber();
132 				PSNumber tx = os.popNumber();
133 				a.set(0, 1);
134 				a.set(1, 0);
135 				a.set(2, 0);
136 				a.set(3, 1);
137 				a.set(4, tx);
138 				a.set(5, ty);
139 				os.push(a);
140 			} else {
141 				error(os, new RangeCheck());
142 			}
143 		} else {
144 			error(os, new TypeCheck());
145 		}
146 		return true;
147 	}

183 class Rotate extends MatrixOperator {
184 	{
185 		operandTypes = new Class[] { PSObject.class };
186 	}
187 
188 	@Override
189 	public boolean execute(OperandStack os) {
190 		if (os.checkType(PSNumber.class)) {
191 			double angle = os.popNumber().getDouble();
192 			os.gstate().rotate(Math.toRadians(angle));
193 		} else if (os.checkType(PSNumber.class, PSArray.class)) {
194 			PSArray a = os.popArray();
195 			if (a.size() >= 6) {
196 				double theta = Math.toRadians(os.popNumber().getDouble());
197 				double sinTheta = Math.sin(theta);
198 				double cosTheta = Math.cos(theta);
199 				a.set(0, cosTheta);
200 				a.set(1, sinTheta);
201 				a.set(2, -sinTheta);
202 				a.set(3, cosTheta);
203 				a.set(4, 0);
204 				a.set(5, 0);
205 				os.push(a);
206 			} else {
207 				error(os, new RangeCheck());
208 			}
209 		} else {
210 			error(os, new TypeCheck());
211 		}
212 		return true;
213 	}

251 class Transform extends MatrixOperator {
252 	{
253 		operandTypes = new Class[] { PSObject.class };
254 	}
255 
256 	@Override
257 	public boolean execute(OperandStack os) {
258 		AffineTransform transform;
259 		double dx, dy;
260 		if (os.checkType(PSNumber.class, PSNumber.class)) {
261 			dy = os.popNumber().getDouble();
262 			dx = os.popNumber().getDouble();
263 			transform = os.gstate().getTransform();
264 		} else if (os.checkType(PSNumber.class, PSNumber.class,
265 				PSPackedArray.class)) {
266 			double[] m = os.popPackedArray().toDoubles();
267 			dy = os.popNumber().getDouble();
268 			dx = os.popNumber().getDouble();
269 			transform = new AffineTransform(m);
270 		} else {
271 			error(os, new TypeCheck());
272 			return true;
273 		}
274 		Point2D d = transform.transform(new Point2D.Double(dx, dy), null);
275 		os.push(d.getX());
276 		os.push(d.getY());
277 		return true;
278 	}
