84  class RectStroke extends PaintingOperator {
85  	{
86  		operandTypes = new Class[] { PSObject.class };
87  	}
88  
89  	private static void stroke(OperandStack os, double x, double y, double w,
90  			double h, AffineTransform m) {
91  		Rectangle2D r = new Rectangle2D.Double(x, y, w, h);
92  		os.gstate().stroke(r, m);
93  	}
94  
95  	private static void stroke(OperandStack os, PSPackedArray a,
96  			AffineTransform m) {
97  		for (int i = 0; i < a.size() / 4; i++) {
98  			double x = ((PSNumber) a.get(i * 4)).getDouble();
99  			double y = ((PSNumber) a.get(i * 4 + 1)).getDouble();
100 			double w = ((PSNumber) a.get(i * 4 + 2)).getDouble();
101 			double h = ((PSNumber) a.get(i * 4 + 3)).getDouble();
102 			stroke(os, x, y, w, h, m);
103 		}
104 	}
105 
106 	@Override
107 	public boolean execute(OperandStack os) {
108 		if (os.checkType(PSNumber.class, PSNumber.class, PSNumber.class,
109 				PSNumber.class)) {
110 			double h = os.popNumber().getDouble();
111 			double w = os.popNumber().getDouble();
112 			double y = os.popNumber().getDouble();
113 			double x = os.popNumber().getDouble();
114 			stroke(os, x, y, w, h, null);
115 		} else if (os.checkType(PSNumber.class, PSNumber.class, PSNumber.class,
116 				PSNumber.class, PSPackedArray.class)) {
117 			AffineTransform m = new AffineTransform(os.popPackedArray()
118 					.toDoubles());
119 			double h = os.popNumber().getDouble();
120 			double w = os.popNumber().getDouble();
121 			double y = os.popNumber().getDouble();
122 			double x = os.popNumber().getDouble();
123 			stroke(os, x, y, w, h, m);
124 		} else if (os.checkType(PSPackedArray.class, PSPackedArray.class)) {
125 			AffineTransform m = new AffineTransform(os.popPackedArray()
126 					.toDoubles());
127 			PSPackedArray a = os.popPackedArray();
128 			stroke(os, a, m);
129 		} else if (os.checkType(PSPackedArray.class)) {
130 			PSPackedArray a = os.popPackedArray();
131 			stroke(os, a, null);
132 		} else {
133 			// FIXME: encoded strings not handled
134 			error(os, new TypeCheck());
135 		}
136 		return true;
137 	}

173 class UStroke extends PaintingOperator {
174 	private boolean done;
175 	private AffineTransform matrix;
176 
177 	private UStroke(boolean d, AffineTransform m) {
178 		done = d;
179 		matrix = m;
180 	}
181 
182 	public UStroke() {
183 	}
184 
185 	@Override
186 	public boolean execute(OperandStack os) {
187 		if (!done) {
188 			if (!os.checkType(PSPackedArray.class)) {
189 				error(os, new TypeCheck());
190 				return true;
191 			}
192 
193 			final int matrixSize = 6;
194 			AffineTransform m = null;
195 			PSPackedArray proc = os.popPackedArray();
196 			if (proc.size() == matrixSize) {
197 				try {
198 					m = new AffineTransform(proc.toDoubles());
199 					if (!os.checkType(PSPackedArray.class)) {
200 						error(os, new TypeCheck());
201 						return true;
202 					}
203 					proc = os.popPackedArray();
204 				} catch (ClassCastException e) {
205 					// no matrix
206 				}
207 			}
208 
209 			// gsave, newpath, systemdict, begin
210 			os.gsave();
211 			os.gstate().newPath();
212 			os.dictStack().push(os.dictStack().systemDictionary());
213 			PSPackedArray upath = (PSPackedArray) proc.copy();
214 			upath.setExecutable();
215 
216 			os.execStack().pop();
217 			os.execStack().push(new UStroke(true, m));
218 			os.execStack().push(upath);
219 			return false;
220 		}
221 
222 		// upath was executed, end, stroke and grestore
223 		os.dictStack().pop();
224 		if (matrix != null) {
225 			os.gstate().transform(matrix);
226 		}
227 		os.gstate().stroke();
228 		os.grestore();
229 		return true;
230 	}
231 }

329 class ImageOperator extends PaintingOperator {
330 
331 	private boolean imageMask;
332 	private int width;
333 	private int height;
334 	private AffineTransform matrix;
335 	private PSPackedArray[] proc;
336 	private PSDataSource[] source;
337 	private boolean multi;
338 	private int bitsPerComponent;
339 	private PSPackedArray decode;
340 	private boolean interpolate;
341 
342 	private final int dataSize = 8;
343 	private BufferedImage image;
344 	private int pixelBitStride;
345 	private int scanlineStride;
346 	private int components;
347 	private float[] color;
348 	private int[] pixels = new int[4];
349 	private int x, y, c;
350 	private boolean popString;
351 
352 	protected ImageOperator() {
353 	}
354 
355 	protected ImageOperator(boolean mask, int w, int h, int b, PSPackedArray m,
356 			PSObject[] ds, PSPackedArray d, boolean i) {
357 
358 		imageMask = mask;
359 		width = w;
360 		height = h;
361 		switch (b) {
362 		case 1:
363 		case 2:
364 		case 4:
365 		case 8:
366 			// case 12: // FIXME: probably does not work
367 			bitsPerComponent = b;
368 			break;
369 		default:
370 			throw new IllegalArgumentException();
371 		}
372 		matrix = new AffineTransform(m.toDoubles());
373 
374 		multi = (ds.length > 1);
375 		proc = new PSPackedArray[ds.length];
376 		source = new PSDataSource[ds.length];
377 		if (ds[0] instanceof PSPackedArray) {
378 			for (int k = 0; k < ds.length; k++) {
379 				proc[k] = (PSPackedArray) ds[k];
380 			}
381 		} else if (ds[0] instanceof PSDataSource) {
382 			for (int k = 0; k < ds.length; k++) {
383 				source[k] = (PSDataSource) ds[k];
384 			}
385 		} else {
386 			throw new IllegalArgumentException();
387 		}
388 
389 		decode = d;
390 		components = decode.size() / 2;
391 		color = new float[components];
392 		interpolate = i;
393 
394 		image = new BufferedImage(width, height, BufferedImage.TYPE_INT_ARGB);
395 		pixelBitStride = bitsPerComponent * components;
396 		scanlineStride = (int) Math.ceil(width * pixelBitStride / dataSize);
397 		popString = false;
398 		x = 0;
399 		y = 0;
400 		c = 0;
401 	}
402 
403 	public boolean handleImageParameters(OperandStack os, boolean mask) {
404 		if (os.checkType(PSDictionary.class)) {
405 			PSDictionary dict = os.popDictionary();
406 			try {
407 				int type = dict.getInteger("ImageType");
408 				switch (type) {
409 				case 1:
410 					int w = dict.getInteger("Width");
411 					int h = dict.getInteger("Height");
412 					PSPackedArray m = dict.getPackedArray("ImageMatrix");
413 					PSObject multiDataSources = dict.get("MultipleDataSources");
414 					PSObject[] ds;
415 					if ((multiDataSources != null)
416 							&& ((PSBoolean) multiDataSources).getValue()) {
417 						if (mask) {
418 							error(os, new RangeCheck());
419 							return true;
420 						}
421 						ds = dict.getPackedArray("DataSource").toObjects();
422 					} else {
423 						ds = new PSObject[1];
424 						ds[0] = dict.get("DataSource");
425 					}
426 					int b = dict.getInteger("BitsPerComponent");
427 					if (mask && (b != 1)) {
428 						error(os, new RangeCheck());
429 						return true;
430 					}
431 					PSPackedArray d = dict.getPackedArray("Decode");
432 					if (!mask) {
433 						if (d.size() != os.gstate()
434 								.getNumberOfColorSpaceComponents() * 2) {
435 							error(os, new RangeCheck());
436 							return true;
437 						}
438 					} /*
439 					 * else { // FIXME: check d being 1 0 or 0 1 }
440 					 */
441 					PSBoolean inter = (PSBoolean) dict.get("Interpolate");
442 					boolean i = (inter != null) ? inter.getValue() : false;
443 
444 					ImageOperator imageOperator = new ImageOperator(mask, w, h,
445 							b, m, ds, d, i);
446 					os.execStack().pop();
447 					os.execStack().push(imageOperator);
448 					return false;
449 				default:
450 					// Level 3
451 					error(os, new Undefined());
452 					break;
453 				}
454 			} catch (NullPointerException e) {
455 				log.log(Level.SEVERE, e.getMessage(), e.getStackTrace());
456 				error(os, new Undefined());
457 			} catch (IllegalArgumentException e) {
458 				error(os, new Undefined());
459 			}
460 			return true;
461 
462 		} else if (os.checkType(PSInteger.class, PSInteger.class,
463 				PSObject.class, PSPackedArray.class, PSObject.class)) {
464 			PSObject[] ds = new PSObject[1];
465 			ds[0] = os.popObject();
466 			PSPackedArray m = os.popPackedArray();
467 			int b;
468 			PSPackedArray d;
469 			if (mask) {
470 				b = 1;
471 				if (os.popBoolean().getValue()) {
472 					d = new PSPackedArray(new double[] { 1, 0 });
473 				} else {
474 					d = new PSPackedArray(new double[] { 0, 1 });
475 				}
476 			} else {
477 				// image is always DeviceGray
478 				b = os.popInteger().getValue();
479 				d = new PSPackedArray(new double[] { 0, 1 });
480 			}
481 			int h = os.popInteger().getValue();
482 			int w = os.popInteger().getValue();
483 
484 			try {
485 				ImageOperator imageOperator = new ImageOperator(mask, w, h, b,
486 						m, ds, d, false);
487 				os.execStack().pop();
488 				os.execStack().push(imageOperator);
489 			} catch (IllegalArgumentException e) {
490 				error(os, new TypeCheck());
491 				return true;
492 			} catch (ClassCastException e) {
493 				error(os, new TypeCheck());
494 				return true;
495 			}
496 
497 			return false;
498 		} else {
499 
500 			error(os, new TypeCheck());
501 			return true;
502 		}
503 	}
504 
505 	@Override
506 	public boolean execute(OperandStack os) {
507 		boolean eod = false;
508 
509 		// pop the prepared string
510 		if ((proc[c] != null) && popString) {
511 			// store string
512 			if (!os.checkType(PSString.class)) {
513 				error(os, new TypeCheck());
514 			}
515 			PSString s = os.popString();
516 			if (s.size() <= 0) {
517 				eod = true;
518 			}
519 			source[c] = s;
520 		}
521 
522 		int b = -1;
523 		int[] prevIndex = new int[4];
524 		for (int i = 0; i < prevIndex.length; i++) {
525 			prevIndex[i] = -1;
526 		}
527 		// fill part of the image
528 		try {
529 			// file or string is filled
530 			while ((y < height) && !eod) {
531 				while ((x < width) && !eod) {
532 					while ((c < components) && !eod) {
533 						int bitnum = x * pixelBitStride + c * bitsPerComponent;
534 						int index = y * scanlineStride + bitnum / dataSize;
535 						int shift = dataSize - (bitnum & (dataSize - 1))
536 								- bitsPerComponent;
537 						int s = (multi) ? c : 0;
538 
539 						if (index > prevIndex[s]) {
540 							prevIndex[s] = index;
541 							if (proc[s] != null) {
542 								b = (source[s] != null) ? source[s].read() : -1;
543 								if (b < 0) {
544 									popString = true;
545 									PSPackedArray p = (PSPackedArray) proc[s]
546 											.clone();
547 									os.execStack().push(p);
548 									return false;
549 								}
550 							} else {
551 								b = source[s].read();
552 								if (b < 0) {
553 									if (source[s] instanceof PSString) {
554 										source[s].reset();
555 										b = source[s].read();
556 									} else {
557 										eod = true;
558 									}
559 								}
560 							}
561 						}
562 
563 						int pixel = (b >> shift)
564 								& ((1 << bitsPerComponent) - 1);
565 						float dMin = ((PSNumber) decode.get(c * 2 + 0))
566 								.getFloat();
567 						float dMax = ((PSNumber) decode.get(c * 2 + 1))
568 								.getFloat();
569 						color[c] = dMin
570 								+ (pixel * (dMax - dMin) / ((1 << bitsPerComponent) - 1));
571 						c++;
572 					}
573 					float[] rgb;
574 					float alpha;
575 					if (imageMask) {
576 						// image mask
577 						rgb = PSGState.toRGB(os.gstate().color(), os.gstate()
578 								.colorSpace());
579 						alpha = (color[0] == 0) ? 1.0f : 0.0f;
580 					} else {
581 						// normal image
582 						switch (components) {
583 						case 1:
584 							rgb = PSGState.toRGB(color, "DeviceGray");
585 							break;
586 						case 3:
587 							rgb = PSGState.toRGB(color, "DeviceRGB");
588 							break;
589 						case 4:
590 							rgb = PSGState.toRGB(color, "DeviceCMYK");
591 							break;
592 						default:
593 							error(os, new RangeCheck(getClass().getName()
594 									+ ": expected length [1..4], was "
595 									+ components));
596 							return true;
597 						}
598 						alpha = 1.0f;
599 					}
600 					pixels[0] = (int) (rgb[0] * 255);
601 					pixels[1] = (int) (rgb[1] * 255);
602 					pixels[2] = (int) (rgb[2] * 255);
603 					pixels[3] = (int) (alpha * 255);
604 					image.getRaster().setPixel(x, y, pixels);
605 					c = 0;
606 					x++;
607 				}
608 				x = 0;
609 				y++;
610 			}
611 
612 			// FIXME: ignores interpolate parameter
613 			os.gstate().image(image, matrix);
614 		} catch (IOException e) {
615 			error(os, new IOError(e));
616 		} catch (ClassCastException e) {
617 			error(os, new TypeCheck());
618 		} catch (CloneNotSupportedException e) {
619 			error(os, new Unimplemented());
620 		}
621 
622 		return true;
623 	}
624 }

639 class ColorImage extends ImageOperator {
640 
641 	@Override
642 	public boolean execute(OperandStack os) {
643 		if (!os.checkType(PSBoolean.class, PSInteger.class)) {
644 			error(os, new TypeCheck());
645 			return true;
646 		}
647 
648 		int ncomp = os.popInteger().getValue();
649 		boolean multi = os.popBoolean().getValue();
650 
651 		PSPackedArray d;
652 		switch (ncomp) {
653 		case 1:
654 			d = new PSPackedArray(new float[] { 0, 1 });
655 			break;
656 		case 3:
657 			d = new PSPackedArray(new float[] { 0, 1, 0, 1, 0, 1 });
658 			break;
659 		case 4:
660 			d = new PSPackedArray(new float[] { 0, 1, 0, 1, 0, 1, 0, 1 });
661 			break;
662 		default:
663 			error(os, new RangeCheck());
664 			return true;
665 		}
666 
667 		PSObject[] ds;
