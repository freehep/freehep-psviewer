156 class Arc extends PathOperator {
157 	{
158 		operandTypes = new Class[] { PSNumber.class, PSNumber.class,
159 				PSNumber.class, PSNumber.class, PSNumber.class };
160 	}
161 
162 	@Override
163 	public boolean execute(OperandStack os) {
164 		double a2 = os.popNumber().getDouble();
165 		double a1 = os.popNumber().getDouble();
166 		double r = os.popNumber().getDouble();
167 		double y = os.popNumber().getDouble();
168 		double x = os.popNumber().getDouble();
169 
170 		while (a2 < a1) {
171 			a2 += FULL_CIRCLE;
172 		}
173 
174 		Arc2D arc = new Arc2D.Double();
175 		arc.setArcByCenter(x, y, r, -a1, -(a2 - a1), Arc2D.OPEN);
176 		os.gstate().path().append(arc, true);
177 
178 		return true;
179 	}
180 }

338 class ReversePath extends PathOperator {
339 	private float coord[] = new float[PATH_SIZE];
340 	private Stack<Object> stack;
341 
342 	private void createSubPath(GeneralPath reverse) {
343 		while (!stack.empty()) {
344 			String s = (String) stack.pop();
345 			if (s.equals("moveto")) {
346 				float x = ((Float) stack.pop()).floatValue();
347 				float y = ((Float) stack.pop()).floatValue();
348 				reverse.moveTo(x, y);
349 			} else if (s.equals("lineto")) {
350 				float x = ((Float) stack.pop()).floatValue();
351 				float y = ((Float) stack.pop()).floatValue();
352 				reverse.lineTo(x, y);
353 			} else {
354 				float cx1 = ((Float) stack.pop()).floatValue();
355 				float cy1 = ((Float) stack.pop()).floatValue();
356 				float cx2 = ((Float) stack.pop()).floatValue();
357 				float cy2 = ((Float) stack.pop()).floatValue();
358 				float x = ((Float) stack.pop()).floatValue();
359 				float y = ((Float) stack.pop()).floatValue();
360 				reverse.curveTo(cx1, cy1, cx2, cy2, x, y);
361 			}
362 		}
363 	}
364 
365 	@Override
366 	public boolean execute(OperandStack os) {
367 		stack = new Stack<Object>();
368 		GeneralPath path = (GeneralPath) os.gstate().path().clone();
369 		PathIterator iterator = path.getPathIterator(new AffineTransform());
370 		GeneralPath reverse = os.gstate().newPath();
371 
372 		while (!iterator.isDone()) {
373 			switch (iterator.currentSegment(coord)) {
374 			case PathIterator.SEG_MOVETO:
375 				createSubPath(reverse);
376 				stack.push(new Float(coord[1]));
377 				stack.push(new Float(coord[0]));
378 				break;
379 
380 			case PathIterator.SEG_LINETO:
381 				stack.push("lineto");
382 				stack.push(new Float(coord[1]));
383 				stack.push(new Float(coord[0]));
384 				break;
385 
386 			case PathIterator.SEG_CUBICTO:
387 				stack.push(new Float(coord[1]));
388 				stack.push(new Float(coord[0]));
389 				stack.push(new Float(coord[3]));
390 				stack.push(new Float(coord[2]));
391 				stack.push("curveto");
392 				stack.push(new Float(coord[5]));
393 				stack.push(new Float(coord[4]));
394 				break;
395 
396 			case PathIterator.SEG_CLOSE:
397 				stack.push("moveto");
398 				createSubPath(reverse);
399 				reverse.closePath();
400 				break;
401 
402 			default:
403 				error(os, new RangeCheck());
404 				return true;
405 			}
406 			iterator.next();
407 		}
408 		return true;
409 	}
410 }

421 class UStrokePath extends PathOperator {
422 	private boolean done;
423 	private AffineTransform matrix;
424 
425 	private UStrokePath(boolean d, AffineTransform m) {
426 		done = d;
427 		matrix = m;
428 	}
429 
430 	public UStrokePath() {
431 	}
432 
433 	@Override
434 	public boolean execute(OperandStack os) {
435 		if (!done) {
436 			if (!os.checkType(PSPackedArray.class)) {
437 				error(os, new TypeCheck());
438 				return true;
439 			}
440 
441 			AffineTransform m = null;
442 			PSPackedArray proc = os.popPackedArray();
443 			if (proc.size() == MATRIX_SIZE) {
444 				try {
445 					m = new AffineTransform(proc.toDoubles());
446 					if (!os.checkType(PSPackedArray.class)) {
447 						error(os, new TypeCheck());
448 						return true;
449 					}
450 					proc = os.popPackedArray();
451 				} catch (ClassCastException e) {
452 					// no matrix
453 				}
454 			}
455 
456 			// newpath, systemdict, begin
457 			os.gstate().newPath();
458 			os.dictStack().push(os.dictStack().systemDictionary());
459 			PSPackedArray upath = (PSPackedArray) proc.copy();
460 			upath.setExecutable();
461 
462 			os.execStack().pop();
463 			os.execStack().push(new UStrokePath(true, m));
464 			os.execStack().push(upath);
465 			return false;
466 		}
467 
468 		// upath was executed, end, strokepath
469 		os.dictStack().pop();
470 		if (matrix != null) {
471 			AffineTransform ctm = os.gstate().getTransform();
472 			os.gstate().transform(matrix);
473 			os.gstate().strokePath();
474 			os.gstate().setTransform(ctm);
475 		} else {
476 			os.gstate().strokePath();
477 		}
478 		// FIXME: where is the gsave???
479 		os.grestore();
480 		return true;
481 	}
482 }

484 class CharPath extends PathOperator {
485 	{
486 		operandTypes = new Class[] { PSString.class, PSBoolean.class };
487 	}
488 
489 	@Override
490 	public boolean execute(OperandStack os) {
491 		boolean strokePath = os.popBoolean().getValue();
492 		String text = os.popString().getValue();
493 		PSGState gs = os.gstate();
494 
495 		float x = 0.0f;
496 		float y = 0.0f;
497 		Point2D point = gs.position();
498 		if (point != null) {
499 			x = (float) point.getX();
500 			y = (float) point.getY();
501 		}
502 
503 		for (int i = 0; i < text.length(); i++) {
504 			int ch = text.charAt(i);
505 			Shape outline = gs.charPath(ch, x, y, strokePath);
506 			// FIXME: not necessary if fonts are all implemented in charPath
507 			if (outline != null) {
508 				gs.path().append(outline, false);
509 			}
510 
511 			x += FontOperator.stringWidth(os, ch);
512 			gs.path().moveTo(x, y);
513 		}
514 		return true;
515 	}
516 }

601 class PathForAll extends PathOperator implements LoopingContext {
602 
603 	private Shape path = null;
604 	private PSPackedArray moveProc;
605 	private PSPackedArray lineProc;
606 	private PSPackedArray curveProc;
607 	private PSPackedArray closeProc;
608 	private PathIterator iterator;
609 	private double[] coord;
610 
611 	private PathForAll(Shape shape, PSPackedArray move, PSPackedArray line,
612 			PSPackedArray curve, PSPackedArray close) {
613 		path = shape;
614 		moveProc = move;
615 		lineProc = line;
616 		curveProc = curve;
617 		closeProc = close;
618 		iterator = path.getPathIterator(new AffineTransform());
619 		coord = new double[PATH_SIZE];
620 	}
621 
622 	public PathForAll() {
623 	}
624 
625 	@Override
626 	public boolean execute(OperandStack os) {
627 
628 		if (path == null) {
629 			if (!os.checkType(PSPackedArray.class, PSPackedArray.class,
630 					PSPackedArray.class, PSPackedArray.class)) {
631 				error(os, new TypeCheck());
632 				return true;
633 			}
634 
635 			PSPackedArray close = os.popPackedArray();
636 			PSPackedArray curve = os.popPackedArray();
637 			PSPackedArray line = os.popPackedArray();
638 			PSPackedArray move = os.popPackedArray();
639 			Shape shape = (Shape) os.gstate().path().clone();
640 
641 			os.execStack().pop();
642 			os.execStack()
643 					.push(new PathForAll(shape, move, line, curve, close));
644 			return false;
645 		}
646 
647 		if (iterator.isDone()) {
648 			return true;
649 		}
650 
651 		switch (iterator.currentSegment(coord)) {
652 		case PathIterator.SEG_MOVETO:
653 			os.push(coord[0]);
654 			os.push(coord[1]);
655 			os.execStack().push(moveProc);
656 			break;
657 
658 		case PathIterator.SEG_LINETO:
659 			os.push(coord[0]);
660 			os.push(coord[1]);
661 			os.execStack().push(lineProc);
662 			break;
663 
664 		case PathIterator.SEG_CUBICTO:
665 			os.push(coord[0]);
666 			os.push(coord[1]);
667 			os.push(coord[2]);
668 			os.push(coord[3]);
669 			os.push(coord[4]);
670 			os.push(coord[5]);
671 			os.execStack().push(curveProc);
672 			break;
673 
674 		case PathIterator.SEG_CLOSE:
