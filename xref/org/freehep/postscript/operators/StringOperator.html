9   /**
10   * String Operators for PostScript Processor
11   * 
12   * @author Mark Donszelmann
13   */
14  public abstract class StringOperator extends AbstractOperator {
15  
16  	public static void register(PSDictionary dict) {
17  		AbstractOperator.register(dict, new Class<?>[] { StringString.class,
18  				AnchorSearch.class, Search.class });
19  	}
20  }
21  
22  class StringString extends StringOperator {
23  	{
24  		operandTypes = new Class[] { PSInteger.class };
25  	}
26  
27  	@Override
28  	public String getName() {
29  		return ("string");
30  	}
31  
32  	@Override
33  	public boolean execute(OperandStack os) {
34  		PSInteger n = os.popInteger();
35  		if (n.getValue() < 0) {
36  			error(os, new RangeCheck());
37  		} else {
38  			os.push(new PSString(n.getValue()));
39  		}
40  		return true;
41  	}
42  }

44  class AnchorSearch extends StringOperator {
45  	{
46  		operandTypes = new Class[] { PSString.class, PSString.class };
47  	}
48  
49  	@Override
50  	public boolean execute(OperandStack os) {
51  		PSString seek = os.popString();
52  		PSString string = os.popString();
53  		if (string.indexOf(seek) != 0) {
54  			os.push(string);
55  			os.push(false);
56  		} else {
57  			os.push(string.subString(seek.size()));
58  			os.push(string.subString(0, seek.size()));
59  			os.push(true);
60  		}
61  		return true;
62  	}
63  }

65  class Search extends StringOperator {
66  	{
67  		operandTypes = new Class[] { PSString.class, PSString.class };
68  	}
69  
70  	@Override
71  	public boolean execute(OperandStack os) {
72  		PSString seek = os.popString();
73  		PSString string = os.popString();
74  		int pos = string.indexOf(seek);
75  		if (pos < 0) {
76  			os.push(string);
77  			os.push(false);
78  		} else {
79  			os.push(string.subString(pos + seek.size()));
80  			os.push(string.subString(pos, seek.size()));
81  			os.push(string.subString(0, pos));
82  			os.push(true);
83  		}
84  		return true;
85  	}
