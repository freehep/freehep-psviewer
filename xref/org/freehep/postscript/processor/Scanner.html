62  	public PSObject nextToken(boolean packingMode, NameLookup lookup)
63  			throws IOException, SyntaxException, NameNotFoundException {
64  		PSObject obj = nextObject(packingMode, lookup);
65  		// deferred execution ?
66  		while ((openCurly > 0) && (!(obj instanceof PSComment))
67  				&& (!(obj instanceof PSDSC))) {
68  			procStack.push(obj);
69  			obj = nextObject(packingMode, lookup);
70  		}
71  		return obj;
72  	}

74  	private PSObject nextObject(boolean packingMode, NameLookup lookup)
75  			throws IOException, SyntaxException, NameNotFoundException {
76  
77  		buffer.reset();
78  
79  		// needed in case DSC reads too far searching for continuation DSC
80  		// comments.
81  		if (nextIsComment) {
82  			return readComment();
83  		}
84  		if (nextIsDSC) {
85  			return readDSC();
86  		}
87  
88  		int next = readNext();
89  		switch (next) {
90  		case -1:
91  			return null;
92  
93  		case '(':
94  			return readString();
95  
96  		case ')':
97  			throw new SyntaxException(lineNo, "Unbalanced Parenthesis");
98  
99  		case '~':
100 			next = in.read();
101 			if (next == '>') {
102 				throw new SyntaxException(lineNo, "Non matching '~>'");
103 			}
104 			if (next > 0) {
105 				in.unread(next);
106 			}
107 			return readName('~', false);
108 
109 		case '>':
110 			next = in.read();
111 			if (next == '>') {
112 				return new PSName(">>");
113 			}
114 			if (next > 0) {
115 				in.unread(next);
116 			}
117 			throw new SyntaxException(lineNo, "Non matching '>'");
118 
119 		case '<':
120 			return readEncoding();
121 
122 		case '[':
123 			openSquare++;
124 			return new PSName("[");
125 
126 		case ']':
127 			if (openSquare > 0) {
128 				openSquare--;
129 				return new PSName("]");
130 			}
131 			throw new SyntaxException(lineNo, "Non matching ']'");
132 
133 		case '{':
134 			openCurly++;
135 			return new PSMark();
136 
137 		case '}':
138 			if (openCurly > 0) {
139 				openCurly--;
140 			}
141 
142 			int n = procStack.countToMark();
143 			if (n < 0) {
144 				throw new SyntaxException(lineNo, "Unmatched '{'");
145 			}
146 
147 			PSObject[] array = new PSObject[n];
148 			for (int i = n - 1; i >= 0; i--) {
149 				array[i] = procStack.popObject();
150 			}
151 			procStack.popMark();
152 
153 			PSPackedArray proc;
154 			if (packingMode) {
155 				proc = new PSPackedArray(array);
156 			} else {
157 				proc = new PSArray(array);
158 			}
159 			proc.setExecutable();
160 
161 			return proc;
162 
163 		case '/':
164 			return readLiteral(lookup);
165 
166 		case '%': // normal comment or DSC
167 			next = in.read();
168 			if (((next == '%') || (next == '!')) && (dsc != null)) {
169 				return readDSC();
170 			}
171 			if (next > 0) {
172 				in.unread(next);
173 			}
174 			return readComment();
175 
176 		case '@': // special name we do not handle at this time, normally
177 			// associated to PCL commands
178 			// FIXME, look up the PCL details somewhere.
179 			return readComment();
180 
181 		case '+':
182 		case '-':
183 		case '.':
184 		case '0':
185 		case '1':
186 		case '2':
187 		case '3':
188 		case '4':
189 		case '5':
190 		case '6':
191 		case '7':
192 		case '8':
193 		case '9':
194 			return readNumber(next);
195 
196 		default:
197 			return readName(next, false);
198 		}
199 	}

205 	private PSString readString() throws IOException {
206 		int balance = 1;
207 		int next = in.read();
208 		;
209 		while ((balance > 0) && (next != -1)) {
210 			switch (next) {
211 			case '\\':
212 				escapeChar();
213 				break;
214 
215 			case '(':
216 				balance++;
217 				buffer.append(next);
218 				break;
219 
220 			case ')':
221 				balance--;
222 				if (balance > 0) {
223 					buffer.append(next);
224 				}
225 				break;
226 
227 			case '\r': // replaced by LF in string
228 				next = in.read();
229 				if ((next != '\n') && (next > 0)) {
230 					in.unread(next);
231 				}
232 				buffer.append('\n');
233 				lineNo++;
234 				break;
235 
236 			case '\n':
237 				buffer.append(next);
238 				lineNo++;
239 				break;
240 
241 			default:
242 				buffer.append(next);
243 				break;
244 			}
245 			next = in.read();
246 		}
247 
248 		if ((balance == 0) && (next > 0)) {
249 			in.unread(next);
250 		}
251 		return new PSString(buffer.getChars());
252 	}

254 	private void escapeChar() throws IOException {
255 		int next = in.read();
256 		if (next == -1) {
257 			throw new EOFException("Unexpected End Of File");
258 		}
259 
260 		switch (next) {
261 		case 'n':
262 			buffer.append('\n');
263 			break;
264 		case 'r':
265 			buffer.append('\r');
266 			break;
267 		case 't':
268 			buffer.append('\t');
269 			break;
270 		case 'b':
271 			buffer.append('\b');
272 			break;
273 		case 'f':
274 			buffer.append('\f');
275 			break;
276 		case '\\':
277 			buffer.append('\\');
278 			break;
279 		case '0':
280 		case '1':
281 		case '2':
282 		case '3':
283 		case '4':
284 		case '5':
285 		case '6':
286 		case '7':
287 			StringBuffer octal = new StringBuffer(3);
288 			octal.append(next - '0');
289 			next = in.read();
290 			if (next == -1) {
291 				throw new EOFException("Unexpected End Of File");
292 			}
293 			if (('0' <= next) && (next <= '7')) {
294 				octal.append(next - '0');
295 				next = in.read();
296 				if (next == -1) {
297 					throw new EOFException("Unexpected End Of File");
298 				}
299 				if (('0' <= next) && (next <= '7')) {
300 					octal.append(next - '0');
301 					// next = in.read();
302 				} else if (next > 0) {
303 					in.unread(next);
304 				}
305 			} else if (next > 0) {
306 				in.unread(next);
307 			}
308 			buffer.append(Integer.parseInt(octal.toString(), 8));
309 			return;
310 
311 		case '\n':
312 			next = in.read();
313 			lineNo++;
314 			return;
315 
316 		case '\r':
317 			next = in.read();
318 			if (next == '\n') {
319 				next = in.read();
320 			}
321 			lineNo++;
322 			return;
323 
324 		case '(':
325 		case ')':
326 		default:
327 			buffer.append(next);
328 			break;
329 		}
330 	}

332 	private PSComment readComment() throws IOException {
333 		nextIsComment = false;
334 
335 		int next = in.read();
336 		while ((next != -1) && (next != '\r') && (next != '\n')) {
337 			buffer.append(next);
338 			next = in.read();
339 		}
340 
341 		return new PSComment(buffer.getString());
342 	}

344 	private PSDSC readDSC() throws IOException {
345 		nextIsDSC = false;
346 
347 		while (true) {
348 			int next = in.read();
349 			while ((next != -1) && (next != '\r') && (next != '\n')) {
350 				buffer.append(next);
351 				next = in.read();
352 			}
353 
354 			// read next ignoring '\r'
355 			next = in.read();
356 			if (next == '\r') {
357 				next = in.read();
358 			}
359 
360 			if (next != '%') {
361 				if (next > 0) {
362 					in.unread(next);
363 				}
364 				break;
365 			}
366 
367 			next = in.read();
368 			if (next != '%') {
369 				if (next > 0) {
370 					in.unread(next);
371 				}
372 				nextIsComment = true;
373 				break;
374 			}
375 
376 			next = in.read();
377 			if (next != '+') {
378 				if (next > 0) {
379 					in.unread(next);
380 				}
381 				nextIsDSC = true;
382 				break;
383 			}
384 		}
385 
386 		return new PSDSC(buffer.getString(), dsc);
387 	}

389 	private PSObject readLiteral(NameLookup lookup) throws IOException,
390 			SyntaxException, NameNotFoundException {
391 		int next = in.read();
392 		if (next == -1) {
393 			throw new IOException("Unexpected EOF reached");
394 		}
395 
396 		if (next == '/') {
397 			// immediate
398 			next = in.read();
399 			PSName name = readName(next, true);
400 			PSObject obj = lookup.lookup(name);
401 			if (obj == null) {
402 				throw new NameNotFoundException("Immediate name /'" + name
403 						+ " not found.");
404 			}
405 			return obj;
406 		}
407 		return readName(next, true);
408 	}

410 	private PSName readName(int next, boolean literal) throws IOException,
411 			SyntaxException {
412 		boolean found = false;
413 		while (!found) {
414 			switch (next) {
415 			case -1:
416 			case '\000':
417 			case '\b':
418 			case '\t':
419 			case '\n':
420 			case '\f':
421 			case '\r':
422 			case ' ':
423 				found = true;
424 				break;
425 
426 			case ')':
427 				throw new SyntaxException(lineNo, "Unbalanced Parenthesis");
428 
429 			case '%':
430 			case '(':
431 			case '/':
432 			case '<':
433 			case '>':
434 			case '[':
435 			case ']':
436 			case '{':
437 			case '}':
438 				found = true;
439 				in.unread(next);
440 				break;
441 
442 			default:
443 				buffer.append(next);
444 				next = in.read();
445 				break;
446 			}
447 		}
448 
449 		PSName name = new PSName(buffer.getString(), literal);
450 		return name;
451 	}

453 	private PSObject readNumber(int next) throws IOException, SyntaxException {
454 		String name = readName(next, false).getValue();
455 		try {
456 			return PSUtils.parseNumber(name);
457 		} catch (NumberFormatException e) {
458 		}
459 
460 		return new PSName(name);

463 	private PSObject readEncoding() throws IOException, SyntaxException {
464 		int next = in.read();
465 		switch (next) {
466 		case '~':
467 			ASCII85InputStream in85 = new ASCII85InputStream(in);
468 			try {
469 				next = in85.read();
470 				while (next != -1) {
471 					buffer.append(next);
472 					next = in85.read();
473 				}
474 			} catch (EncodingException e) {
475 				SyntaxException se = new SyntaxException(lineNo, e.getMessage());
476 				se.initCause(e);
477 				throw se;
478 			}
479 			lineNo += in85.getLineNo() - 1;
480 			return new PSString(buffer.getChars());
481 
482 		case '<':
483 			return new PSName("<<");
484 
485 		default:
486 			if (next > 0) {
487 				in.unread(next);
488 			}
489 			ASCIIHexInputStream inHex = new ASCIIHexInputStream(in);
490 			try {
491 				next = inHex.read();
492 				while (next != -1) {
493 					buffer.append(next);
494 					next = inHex.read();
495 				}
496 			} catch (EncodingException e) {
497 				SyntaxException se = new SyntaxException(lineNo, e.getMessage());
498 				se.initCause(e);
499 				throw se;
500 			}
501 			lineNo += inHex.getLineNo() - 1;
502 			return new PSString(buffer.getChars());
503 		}
504 	}

506 	// skips whitespace until next token
507 	private int readNext() throws IOException {
508 		boolean ws;
509 		int next;
510 		do {
511 			next = in.read();
512 			switch (next) {
513 			case '\r':
514 				next = in.read();
515 				if ((next != '\n') && (next > 0)) {
516 					in.unread(next);
517 				}
518 				// fallthrough
519 			case '\n':
520 				lineNo++;
521 				ws = true;
522 				break;
523 			case '\000':
524 			case '\b':
525 			case '\t':
526 			case '\f':
527 			case ' ':
528 				ws = true;
529 				break;
530 			default:
531 				ws = false;
532 			}
533 		} while ((next != -1) && ws);
534 
535 		return (ws) ? -1 : next;
536 	}

538 	public static void main(String[] args) throws Exception {
539 		if (args.length != 1) {
540 			log.info("Usage: Scanner filename.ps");
541 			System.exit(1);
542 		}
543 
544 		DSC dsc = new DSC();
545 		Scanner scanner = new Scanner(new PushbackInputStream(
546 				new FileInputStream(args[0])), dsc);
547 
548 		PSObject token;
549 		while ((token = scanner.nextToken(true, null)) != null) {
550 			log.info(scanner.getLineNo() + ": "
551 					+ token.getClass().getName() + ": " + token);
552 		}
553 
554 		scanner.close();
555 	}
