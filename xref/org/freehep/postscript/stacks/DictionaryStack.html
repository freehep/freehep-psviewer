214 	public DictionaryStack() {
215 		super();
216 
217 		// fill error dictionary
218 		PSDictionary error = new PSDictionary();
219 		error.setName("errordict");
220 		ErrorOperator.register(error);
221 
222 		// fill $error dictionary
223 		PSDictionary dollarErrorDict = new PSDictionary();
224 		dollarErrorDict.setName("$error");
225 		dollarErrorDict.put("newerror", new PSBoolean(false));
226 		dollarErrorDict.put("errorname", new PSNull());
227 		dollarErrorDict.put("command", new PSNull());
228 		dollarErrorDict.put("errorinfo", new PSNull());
229 		dollarErrorDict.put("ostack", new PSNull());
230 		dollarErrorDict.put("estack", new PSNull());
231 		dollarErrorDict.put("dstack", new PSNull());
232 		dollarErrorDict.put("recordstacks", new PSBoolean(true));
233 		dollarErrorDict.put("binary", new PSBoolean(false));
234 
235 		PSDictionary user = new PSDictionary();
236 		user.setName("userdict");
237 
238 		PSDictionary global = new PSDictionary();
239 		global.setName("globaldict");
240 
241 		PSDictionary status = new PSDictionary();
242 		status.setName("statusdict");
243 
244 		// fill system dictionary
245 		PSDictionary system = new PSDictionary();
246 		system.setName("systemdict");
247 		system.put(systemdict, system);
248 		system.put(errordict, error);
249 		system.put(dollarerror, dollarErrorDict);
250 		system.put(userdict, user);
251 		system.put(statusdict, status);
252 		system.put(globaldict, global);
253 
254 		// register all operator groups
255 		GeneralOperator.register(system);
256 		StackOperator.register(system);
257 		ArithmeticOperator.register(system);
258 		ArrayOperator.register(system);
259 		PackedArrayOperator.register(system);
260 		DictionaryOperator.register(system);
261 		StringOperator.register(system);
262 		RelationalOperator.register(system);
263 		ControlOperator.register(system);
264 		ConversionOperator.register(system);
265 		FileOperator.register(system);
266 		GraphicsStateOperator.register(system);
267 		DeviceOperator.register(system);
268 		MatrixOperator.register(system);
269 		MemoryOperator.register(system);
270 		MiscellaneousOperator.register(system);
271 		PathOperator.register(system);
272 		PaintingOperator.register(system);
273 		FormOperator.register(system);
274 		OutputOperator.register(system);
275 		FontOperator.register(system);
276 		ExtraOperator.register(system);
277 
278 		// add encodings
279 		system.put(standardEncoding, standardEncodingArray);
280 		system.put(isoLatin1Encoding, isoLatin1EncodingArray);
281 		system.put(symbolEncoding, symbolEncodingArray);
282 		system.put(zapfDingbatsEncoding, zapfDingbatsEncodingArray);
283 
284 		// add font directories
285 		system.put(fontDirectory, new PSDictionary());
286 		// FIXME: should be a different directory
287 		system.put(globalFontDirectory, system.get(fontDirectory));
288 
289 		// others
290 		system.put(new PSName("null"), new PSNull());
291 		system.put(new PSName("languagelevel"), new PSInteger(2));
292 
293 		// create dictionary stack
294 		push(system);
295 		push(global);
296 		push(user);
297 	}
