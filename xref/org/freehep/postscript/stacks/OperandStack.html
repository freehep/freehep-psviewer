196 	public void roll(int n, int j) {
197 		// make j positive
198 		if (j < 0) {
199 			j = n + j;
200 		}
201 
202 		// skip if unnecessary
203 		if ((j == n) || (j == 0)) {
204 			return;
205 		}
206 
207 		// move to tmp
208 		Object[] tmp = new Object[j];
209 		for (int i = 0; i < j; i++) {
210 			tmp[i] = elementData[elementCount - j + i];
211 		}
212 
213 		// move pre
214 		for (int i = n - j - 1; i >= 0; i--) {
215 			elementData[elementCount - n + j + i] = elementData[elementCount
216 					- n + i];
217 		}
218 
219 		// move post
220 		for (int i = 0; i < j; i++) {
221 			elementData[elementCount - n + i] = tmp[i];
222 		}
223 	}

225 	public boolean clearToMark() {
226 		int i = elementCount - 1;
227 		if (i < 0) {
228 			return false;
229 		}
230 
231 		while (i >= 0 && !(elementData[i] instanceof PSMark)) {
232 			i--;
233 		}
234 
235 		if (i < 0) {
236 			return false;
237 		}
238 		elementCount = i;
239 		return true;
240 	}

309 	public boolean checkType(Class<?>[] types) {
310 		if (elementCount < types.length) {
311 			log.warning("Found " + elementCount + " while expecting "
312 					+ types.length);
313 			return false;
314 		}
315 
316 		for (int i = 0; i < types.length; i++) {
317 			Object obj = elementData[elementCount - types.length + i];
318 			if (!types[i].isInstance(obj)) {
319 				log.warning("Found " + obj + " of " + obj.getClass()
320 						+ " while expecting " + types[i]);
321 				return false;
322 			}
323 		}
324 		return true;
325 	}
