4   import java.util.Stack;
5   import java.util.logging.Logger;
6   
7   import org.freehep.postscript.types.PSArray;
8   import org.freehep.postscript.types.PSMark;
9   import org.freehep.postscript.types.PSName;
10  import org.freehep.postscript.types.PSObject;

12  /**
13   * PostScript Abstract Stack for PostScript Processor
14   * 
15   * @author Mark Donszelmann
16   */
17  public class PostScriptStack extends Stack<Object> {
18  	protected Logger log = Logger.getLogger("org.freehep.postscript");
19  
20  	protected int marker = -1;
21  
22  	@Override
23  	public Object push(Object o) {
24  		if (o == null) {
25  			throw new IllegalArgumentException(
26  					"PostScript Stack cannot contain 'null'");
27  		}
28  
29  		if (!(o instanceof PSObject)) {
30  			throw new IllegalArgumentException(
31  					"PostScript Stack can only contain PSObjects, not: " + o);
32  		}
33  
34  		return super.push(o);
35  	}
36  
37  	public Object push(String s) {
38  		return super.push(new PSName(s));
39  	}
40  
41  	public PSObject peekObject() {
42  		return (PSObject) peek();
43  	}
44  
45  	public PSObject peekObject(int i) {
46  		if (i > elementCount) {
47  			return null;
48  		}
49  		return (PSObject) elementData[elementCount - i - 1];
50  	}
51  
52  	// do not remove elements, just lower the top of the stack
53  	@Override
54  	public Object pop() {
55  		if (elementCount == 0) {
56  			return null;
57  		}
58  
59  		elementCount--;
60  		return elementData[elementCount];
61  	}
62  
63  	public PSObject popObject() {
64  		return (PSObject) pop();
65  	}
66  
67  	// mark and counttomark needed for making procedure objects
68  	public int countToMark() {
69  		int count = 0;
70  		int i = elementCount - 1;
71  		while (i > 0 && !(elementData[i] instanceof PSMark)) {
72  			i--;
73  			count++;
74  		}
75  		return (elementData[i] instanceof PSMark) ? count : -1;
76  	}
77  
78  	public PSMark popMark() {
79  		return (PSMark) super.pop();
80  	}
81  
82  	// sets the marker, and cleans up what is above
83  	public void mark() {
84  		marker = elementCount;
85  		setSize(marker);
86  	}
87  
88  	// resets the stack to the marker level, and cleans above
89  	public void reset() {
90  		if (marker >= 0) {
91  			setSize(marker);
92  			marker = -1;
93  		}
94  	}
95  
96  	public void copyInto(PSArray a) {
97  		for (int i = 0; i < a.size(); i++) {
98  			a.set(i, (PSObject) elementData[i]);
99  		}
100 	}
101 
102 	public void printStack() {
103 		for (int i = elementCount - 1; i >= 0; i--) {
104 			log.info(i + ": " + elementData[i].getClass().getName()
105 					+ ": " + elementData[i].toString());
106 		}
107 	}
108 }
