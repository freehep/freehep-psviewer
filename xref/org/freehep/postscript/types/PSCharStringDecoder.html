38  	public PSGlyph decode(PSString charString) throws IOException {
39  
40  		decryption = new EEXECDecryption(charString.getInputStream(),
41  				EEXECConstants.CHARSTRING_R, EEXECConstants.N);
42  		currentX = 0;
43  		currentY = 0;
44  		charProc = new LinkedList<PSObject>();
45  
46  		boolean end = false;
47  		do {
48  			int byte1 = decryption.read();
49  			if (byte1 < 0) {
50  				end = true;
51  			} else if (byte1 >= 0 && byte1 <= 31) {
52  				// command
53  				if (byte1 != 12) {
54  					switch (byte1) {
55  					case 14:
56  						addCommand("eofill");
57  						break;
58  					case 13:
59  						hsbw();
60  						break;
61  					case 9:
62  						addCommand("closepath");
63  						break;
64  					case 6:
65  						addCommand("hlineto");
66  						break;
67  					case 22:
68  						addCommand("hmoveto");
69  						break;
70  					case 31:
71  						addCommand("hcurveto");
72  						break;
73  					case 5:
74  						addCommand("rlineto");
75  						break;
76  					case 21:
77  						addCommand("rmoveto");
78  						break;
79  					case 8:
80  						addCommand("rrcurveto");
81  						break;
82  					case 30:
83  						addCommand("vhcurveto");
84  						break;
85  					case 7:
86  						addCommand("vlineto");
87  						break;
88  					case 4:
89  						addCommand("vmoveto");
90  						break;
91  					case 1:
92  						ignore(2);
93  						break; // hstem
94  					case 3:
95  						ignore(2);
96  						break; // vstem
97  					default:
98  						log.warning("Command " + byte1
99  								+ " not implemented");
100 						break;
101 					}
102 				} else {
103 					int byte2 = decryption.read();
104 					switch (byte2) {
105 					case 7:
106 						sbw();
107 						break;
108 					case 0:
109 						ignore(2);
110 						break; // dotsection
111 					case 2:
112 						ignore(6);
113 						break; // hstem3
114 					case 1:
115 						ignore(6);
116 						break; // vstem3
117 					case 33:
118 						addCommand("setcurrentpoint");
119 						break; // vstem3
120 					default:
121 						log.warning("Command 12 " + byte2 + " not implemented");
122 						break;
123 					}
124 				}
125 			} else {
126 				int value = readNumber(byte1);
127 				addNumber(value);
128 			}
129 		} while (!end);
130 
131 		PSObject[] obj = new PSObject[charProc.size()];
132 		Iterator<PSObject> i = charProc.iterator();
133 		int j = 0;
134 		while (i.hasNext()) {
135 			obj[j++] = i.next();
136 		}
137 
138 		array = new PSPackedArray(obj);
139 		array.setExecutable();
140 
141 		return new PSType1Glyph(array, widthX, sbX);
142 	}

144 	private int readNumber(int v) throws IOException {
145 		int value = 0;
146 		if (v >= 32 && v <= 246) {
147 			value = v - 139;
148 		} else if (v >= 247 && v <= 250) {
149 			int w = decryption.read();
150 			value = ((v - 247) * 256) + w + 108;
151 		} else if (v >= 251 && v <= 254) {
152 			int w = decryption.read();
153 			value = -((v - 251) * 256) - w - 108;
154 		} else { // v == 255
155 			value = decryption.read() << 3 * 8 + decryption.read() << 2 * 8 + decryption
156 					.read() << 1 * 8 + decryption.read() << 0 * 8;
157 		}
158 		return value;
159 	}

161 	private void addCommand(String name) {
162 		charProc.add(dict.get(name));
163 	}

165 	private void addNumber(int i) {
166 		charProc.add(new PSInteger(i));
167 	}

169 	private void ignore(int n) {
170 		for (int i = 0; i < n; i++) {
171 			charProc.remove(charProc.size() - 1);
172 		}
173 	}

175 	private int pop() {
176 		return ((PSInteger) charProc.remove(charProc.size() - 1)).getValue();
177 	}

179 	private void hsbw() {
180 		widthY = 0;
181 		widthX = pop();
182 		sbY = 0;
183 		sbX = pop();
184 		addInitialPoint();
185 	}
186 
187 	private void sbw() {
188 		widthY = pop();
189 		widthX = pop();
190 		sbY = pop();
191 		sbX = pop();
192 		addInitialPoint();
193 	}

195 	private void addInitialPoint() {
196 		if (sbX == 0) {
197 			return;
198 		}
199 		charProc.add(0, dict.get("moveto"));
200 		charProc.add(0, new PSInteger(0));
201 		charProc.add(0, new PSInteger(sbX));
202 	}
