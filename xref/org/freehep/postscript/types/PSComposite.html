9   public abstract class PSComposite extends PSObject {

29

32  	}
33  
34  	public boolean accessExecute() {
35  		return access >= EXECUTE_ONLY;
36  	}
37  
38  	public boolean changeAccess(int newAccess) {
39  		if (newAccess > access) {
40  			return false;
41  		}
42  		access = newAccess;
43  		return true;
44  	}
