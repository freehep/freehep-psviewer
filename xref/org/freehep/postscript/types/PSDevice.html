21  public abstract class PSDevice implements DSCEventListener {
22  
23  	private boolean valid = false;
24  	private AffineTransform mirror = null;
25  	private AffineTransform boundingBox = new AffineTransform();
26  	private AffineTransform pageBoundingBox = null;
27  	private AffineTransform transform = new AffineTransform();
28  	private Graphics2D graphics;
29  
30  	private List<RefreshListener> listeners = new ArrayList<RefreshListener>();
31  
32  	public void addComponentRefreshListener(RefreshListener l) {
33  		listeners.add(l);
34  	}
35  
36  	public void removeComponentRefreshListener(RefreshListener l) {
37  		listeners.remove(l);
38  	}
39  
40  	protected void fireComponentRefreshed() {
41  		valid = false;
42  		mirror = new AffineTransform(1, 0, 0, -1, 0, getHeight());
43  		for (Iterator<RefreshListener> i = listeners.iterator(); i.hasNext();) {
44  			i.next().componentRefreshed();
45  		}
46  	}
47  
48  	public void dscCommentFound(DSCEvent event) {
49  		if (event.getComment().equals("BoundingBox:")) {
50  			Rectangle bb = (Rectangle) event.getArgs();
51  			double s = Math.min(getWidth() / bb.width, getHeight() / bb.height);
52  			boundingBox = new AffineTransform(s, 0, 0, s, -bb.x * s, -bb.y * s);
53  			pageBoundingBox = null;
54  			valid = false;
55  		}
56  
57  		if (event.getComment().equals("PageBoundingBox:")) {
58  			Rectangle bb = (Rectangle) event.getArgs();
59  			double s = Math.min(getWidth() / bb.width, getHeight() / bb.height);
60  			pageBoundingBox = new AffineTransform(s, 0, 0, s, -bb.x * s, -bb.y
61  					* s);
62  			valid = false;
63  		}
64  
65  		if (event.getComment().equals("Page:")) {
66  			pageBoundingBox = null;
67  			valid = false;
68  		}
69  	}
70  
71  	public void setTransform(AffineTransform transform) {
72  		this.transform = (transform != null) ? transform
73  				: new AffineTransform();
74  		valid = false;
75  	}
76  
77  	public AffineTransform getTransform() {
78  		return transform;
79  	}
80  
81  	public AffineTransform getMirrorTransform() {
82  		if (mirror == null) {
83  			mirror = new AffineTransform(1, 0, 0, -1, 0, getHeight());
84  		}
85  		return mirror;
86  	}
87  
88  	public AffineTransform getBoundingBoxTransform() {
89  		return (pageBoundingBox != null) ? pageBoundingBox : boundingBox;
90  	}
91  
92  	// FIXME this method and the next seem clumsy. There must be a way
93  	// for deviceGraphics to remain untouched, and to just add new transforms
94  	// to graphics, BUT getDeviceGraphics().create() does not seem to work,
95  	// presumably because BufferedPanel probably copies a different buffer...
96  	public Graphics2D getGraphics() {
97  		if (!valid) {
98  			graphics = (Graphics2D) getDeviceGraphics();
99  			if (graphics != null) {
100 				graphics.setTransform(getDeviceTransform());
101 				graphics.transform(getTransform());
102 				graphics.transform(getMirrorTransform());
103 				graphics.transform(getBoundingBoxTransform());
104 				valid = true;
105 			}			
106 		}
107 		
108 		return graphics;
109 	}
110 
111 	public void erasePage() {
112 		graphics = (Graphics2D) getDeviceGraphics();
113 		graphics.setBackground(Color.WHITE);
114 		graphics.setTransform(getDeviceTransform());
115 		graphics.clearRect(0, 0, (int) getWidth(), (int) getHeight());
116 		valid = false;
117 	}
