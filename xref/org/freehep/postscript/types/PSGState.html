112 	public void copyInto(PSGState copy) {
113 		copy.device = device;
114 		copy.ctm = (AffineTransform) ctm.clone();
115 		copy.path = (GeneralPath) path.clone();
116 		copy.clipPath = (clipPath == null) ? null : (GeneralPath) clipPath
117 				.clone();
118 		copy.lineWidth = lineWidth;
119 		copy.cap = cap;
120 		copy.join = join;

166 	public Shape charPath(int cc, float x, float y, boolean strokePath) {
167 		Shape charPath;
168 		// FIXME: works only for Java type 1 fonts
169 		switch (font.getInteger("FontType")) {
170 		case 1:
171 			PSName name = font.getPackedArray("Encoding").getName(cc);
172 			PSObject obj = font.getDictionary("CharStrings").get(name);
173 			// FIXME: obj may be PSPackedArray
174 			GlyphVector glyph = ((PSJavaGlyph) obj).getGlyph();
175 			charPath = glyph.getOutline(x, y);
176 			break;
177 
178 		default:
179 			log.warning(getClass() + ": CharPath failed for fonttype: "
180 					+ font.getInteger("FontType"));
181 			return null;
182 			// FontRenderContext fontRenderContext = new FontRenderContext(null,

184 			// GlyphVector boxGlyph = font.createGlyphVector(fontRenderContext,
185 			// '\u25a1');
186 			// return boxGlyph.getOutline(x, y);
187 		}
188 
189 		if (strokePath) {
190 			charPath = new GeneralPath(device.getGraphics().getStroke()
191 					.createStrokedShape(charPath));
192 		}
193 		return charPath;

207 	public void stroke(Shape s, AffineTransform m) {
208 		AffineTransform at = (AffineTransform) ctm.clone();
209 		GeneralPath p = new GeneralPath(s);
210 
211 		if (m != null) {
212 			// apply AT-1 to s
213 			try {
214 				AffineTransform pt = m.createInverse();
215 				p.transform(pt);
216 			} catch (NoninvertibleTransformException e) {
217 				log.warning(Constants.INTERNAL_GSTATE_ERROR);
218 			}
219 
220 			// add new transform
221 			at.concatenate(m);
222 		}
223 
224 		Graphics2D g = (Graphics2D) device.getGraphics().create();
225 		g.transform(at);
226 		g.draw(p);
227 		g.dispose();
228 	}

448 	private static ColorSpace toColorSpace(String name) {
449 		if (name.equals(Constants.DEVICE_GRAY)) {
450 			return ColorSpace.getInstance(ColorSpace.CS_GRAY);
451 		} else if (name.equals(Constants.DEVICE_RGB)) {
452 			return ColorSpace.getInstance(ColorSpace.CS_sRGB);
453 		} else if (name.equals(Constants.DEVICE_CMYK)) {
454 			return ColorSpace.getInstance(ColorSpace.CS_sRGB);
455 		}
456 		return null;
457 	}

463 	public boolean setColorSpace(String name, Object[] params) {
464 		ColorSpace space = toColorSpace(name);
465 		if (space == null) {
466 			if (name.equals(Constants.PATTERN)) {
467 				if (params == null) {
468 					space = new Pattern(toColorSpace(Constants.DEVICE_RGB));
469 				} else {
470 					space = new Pattern(
471 							toColorSpace(((PSName) params[1]).cvs()));
472 				}
473 			}
474 		}
475 
476 		if (space != null) {
477 			colorSpaceName = name;
478 			colorSpace = space;
479 			return true;
480 		}
481 		return false;
482 	}

518 	public void setColor(Paint paint, Object[] params) {
519 		if (params != null && params.length > 0) {
520 			// rebuild paint with new color
521 			Color c;
522 			switch (params.length) {
523 			case 1:
524 				float d = ((PSNumber)params[0]).getFloat();
525 				c = new Color(d, d, d);
526 				break;
527 			case 3:
528 				float r = ((PSNumber)params[0]).getFloat();
529 				float g = ((PSNumber)params[0]).getFloat();
530 				float b = ((PSNumber)params[0]).getFloat();
531 				c = new Color(r, g, b);
532 				break;
533 			default:
534 				log.warning("Number of params not handled " + params.length);
535 				c = Color.BLACK;
536 				break;
537 			}
538 			paint = ((FixedTexturePaint) paint).inColor(c);
539 		}
540 		device.getGraphics().setPaint(paint);
541 	}

552 	public static float[] toRGB(float[] color, String space) {
553 		if (space.equals(Constants.DEVICE_GRAY)) {
554 			float c = Math.max(0.0f, Math.min(1.0f, color[0]));
555 			return new float[] { c, c, c };
556 
557 		} else if (space.equals(Constants.DEVICE_RGB)) {
558 			float r = Math.max(0.0f, Math.min(1.0f, color[0]));
559 			float g = Math.max(0.0f, Math.min(1.0f, color[1]));
560 			float b = Math.max(0.0f, Math.min(1.0f, color[2]));
561 			return new float[] { r, g, b };
562 
563 		} else if (space.equals(Constants.DEVICE_CMYK)) {
564 			float r = Math
565 					.max(0.0f, 1.0f - Math.min(1.0f, color[0] + color[3]));
566 			float g = Math
567 					.max(0.0f, 1.0f - Math.min(1.0f, color[1] + color[3]));
568 			float b = Math
569 					.max(0.0f, 1.0f - Math.min(1.0f, color[2] + color[3]));
570 			return new float[] { r, g, b };
571 		}
572 		return null;
573 	}

579 	public float[] color(String space) {
580 		float[] rgb;
581 		rgb = device.getGraphics().getColor().getColorComponents(null);
582 
583 		if (space.equals(Constants.PATTERN)) {
584 			rgb = colorSpace.toRGB(rgb);
585 			space = toColorSpaceName(colorSpace);
586 		}
587 
588 		rgb = toColor(space, rgb);
589 		if (rgb == null) {
590 			log.warning("Unknown colorspace: " + colorSpaceName);
591 		}
592 		return rgb;
593 	}

595 	public static float[] toColor(String space, float[] rgb) {
596 		float[] color;
597 		if (space.equals(Constants.DEVICE_GRAY)) {
598 			color = new float[1];
599 			color[0] = (float) (0.30 * rgb[0] + 0.59 * rgb[1] + 0.11 * rgb[2]);
600 			return color;
601 
602 		} else if (space.equals(Constants.DEVICE_RGB)) {
603 			return rgb;
604 
605 		} else if (space.equals(Constants.DEVICE_CMYK)) {
606 			float c = 1.0f - rgb[0];
607 			float m = 1.0f - rgb[1];
608 			float y = 1.0f - rgb[2];
609 			float k = Math.min(c, Math.min(m, y));
610 
611 			color = new float[4];
612 			color[0] = Math.min(1.0f, Math.max(0.0f, c));
613 			color[1] = Math.min(1.0f, Math.max(0.0f, m));
614 			color[2] = Math.min(1.0f, Math.max(0.0f, y));
615 			color[3] = Math.min(1.0f, Math.max(0.0f, k));
616 			return color;
617 		} else {
618 			return null;
619 		}
