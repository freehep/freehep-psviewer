112 		copy.device = device;
113 		copy.ctm = (AffineTransform) ctm.clone();
114 		copy.path = (GeneralPath) path.clone();
115 		copy.clipPath = (clipPath == null) ? null : (GeneralPath) clipPath

165 	public Shape charPath(int cc, float x, float y, boolean strokePath) {
166 		Shape charPath;
167 		// FIXME: works only for Java type 1 fonts
168 		switch (font.getInteger("FontType")) {
169 		case 1:
170 			PSName name = font.getPackedArray("Encoding").getName(cc);
171 			PSObject obj = font.getDictionary("CharStrings").get(name);
172 			// FIXME: obj may be PSPackedArray
173 			GlyphVector glyph = ((PSJavaGlyph) obj).getGlyph();
174 			charPath = glyph.getOutline(x, y);
175 			break;
176 
177 		default:
178 			log.warning(getClass() + ": CharPath failed for fonttype: "
179 					+ font.getInteger("FontType"));
180 			return null;
181 			// FontRenderContext fontRenderContext = new FontRenderContext(null,

183 			// GlyphVector boxGlyph = font.createGlyphVector(fontRenderContext,
184 			// '\u25a1');
185 			// return boxGlyph.getOutline(x, y);
186 		}
187 
188 		if (strokePath) {
189 			charPath = new GeneralPath(device.getGraphics().getStroke()
190 					.createStrokedShape(charPath));
191 		}
192 		return charPath;

206 	public void stroke(Shape s, AffineTransform m) {
207 		AffineTransform at = (AffineTransform) ctm.clone();
208 		GeneralPath p = new GeneralPath(s);
209 
210 		if (m != null) {
211 			// apply AT-1 to s
212 			try {
213 				AffineTransform pt = m.createInverse();
214 				p.transform(pt);
215 			} catch (NoninvertibleTransformException e) {
216 				log.warning(Constants.INTERNAL_GSTATE_ERROR);
217 			}
218 
219 			// add new transform
220 			at.concatenate(m);
221 		}
222 
223 		Graphics2D g = (Graphics2D) device.getGraphics().create();
224 		g.transform(at);
225 		g.draw(p);
226 		g.dispose();
227 	}

447 	private static ColorSpace toColorSpace(String name) {
448 		if (name.equals(Constants.DEVICE_GRAY)) {
449 			return ColorSpace.getInstance(ColorSpace.CS_GRAY);
450 		} else if (name.equals(Constants.DEVICE_RGB)) {
451 			return ColorSpace.getInstance(ColorSpace.CS_sRGB);
452 		} else if (name.equals(Constants.DEVICE_CMYK)) {
453 			return ColorSpace.getInstance(ColorSpace.CS_sRGB);
454 		}
455 		return null;
456 	}

462 	public boolean setColorSpace(String name, Object[] params) {
463 		ColorSpace space = toColorSpace(name);
464 		if (space == null) {
465 			if (name.equals(Constants.PATTERN)) {
466 				if (params == null) {
467 					space = new Pattern(toColorSpace(Constants.DEVICE_RGB));
468 				} else {
469 					space = new Pattern(
470 							toColorSpace(((PSName) params[1]).cvs()));
471 				}
472 			}
473 		}
474 
475 		if (space != null) {
476 			colorSpaceName = name;
477 			colorSpace = space;
478 			return true;
479 		}
480 		return false;
481 	}

517 	public void setColor(Paint paint, Object[] params) {
518 		if (params != null && params.length > 0) {
519 			// rebuild paint with new color
520 			Color c;
521 			switch (params.length) {
522 			case 1:
523 				float d = ((PSNumber)params[0]).getFloat();
524 				c = new Color(d, d, d);
525 				break;
526 			case 3:
527 				float r = ((PSNumber)params[0]).getFloat();
528 				float g = ((PSNumber)params[0]).getFloat();
529 				float b = ((PSNumber)params[0]).getFloat();
530 				c = new Color(r, g, b);
531 				break;
532 			default:
533 				log.warning("Number of params not handled " + params.length);
534 				c = Color.BLACK;
535 				break;
536 			}
537 			paint = ((FixedTexturePaint) paint).inColor(c);
538 		}
539 		device.getGraphics().setPaint(paint);
540 	}

551 	public static float[] toRGB(float[] color, String space) {
552 		if (space.equals(Constants.DEVICE_GRAY)) {
553 			float c = Math.max(0.0f, Math.min(1.0f, color[0]));
554 			return new float[] { c, c, c };
555 
556 		} else if (space.equals(Constants.DEVICE_RGB)) {
557 			float r = Math.max(0.0f, Math.min(1.0f, color[0]));
558 			float g = Math.max(0.0f, Math.min(1.0f, color[1]));
559 			float b = Math.max(0.0f, Math.min(1.0f, color[2]));
560 			return new float[] { r, g, b };
561 
562 		} else if (space.equals(Constants.DEVICE_CMYK)) {
563 			float r = Math
564 					.max(0.0f, 1.0f - Math.min(1.0f, color[0] + color[3]));
565 			float g = Math
566 					.max(0.0f, 1.0f - Math.min(1.0f, color[1] + color[3]));
567 			float b = Math
568 					.max(0.0f, 1.0f - Math.min(1.0f, color[2] + color[3]));
569 			return new float[] { r, g, b };
570 		}
571 		return null;
572 	}

578 	public float[] color(String space) {
579 		float[] rgb;
580 		rgb = device.getGraphics().getColor().getColorComponents(null);
581 
582 		if (space.equals(Constants.PATTERN)) {
583 			rgb = colorSpace.toRGB(rgb);
584 			space = toColorSpaceName(colorSpace);
585 		}

589 			log.warning("Unknown colorspace: " + colorSpaceName);
590 		}

594 	public static float[] toColor(String space, float[] rgb) {
595 		float[] color;
596 		if (space.equals(Constants.DEVICE_GRAY)) {
597 			color = new float[1];
598 			color[0] = (float) (0.30 * rgb[0] + 0.59 * rgb[1] + 0.11 * rgb[2]);
599 			return color;
600 
601 		} else if (space.equals(Constants.DEVICE_RGB)) {
602 			return rgb;
603 
604 		} else if (space.equals(Constants.DEVICE_CMYK)) {
605 			float c = 1.0f - rgb[0];
606 			float m = 1.0f - rgb[1];
607 			float y = 1.0f - rgb[2];
608 			float k = Math.min(c, Math.min(m, y));
609 
610 			color = new float[4];
611 			color[0] = Math.min(1.0f, Math.max(0.0f, c));
612 			color[1] = Math.min(1.0f, Math.max(0.0f, m));
613 			color[2] = Math.min(1.0f, Math.max(0.0f, y));
614 			color[3] = Math.min(1.0f, Math.max(0.0f, k));
615 			return color;
616 		} else {
617 			return null;
618 		}
