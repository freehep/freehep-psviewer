60  	public PSInputFile(String filename, int buffer, DSC dsc) throws IOException {
61  		super(filename, false, READ_ONLY);
62  		this.dsc = dsc;
63  		this.buffer = buffer;
64  		init();
65  	}

67  	private void init() throws IOException {
68  		InputStream input;
69  		try {
70  			URL url = new URL(filename);
71  			input = url.openStream();
72  			// FIXME maybe we should modify name here to something shorter...
73  		} catch (MalformedURLException e) {
74  			input = new FileInputStream(filename);
75  		} catch (ZipException e) {
76  			FileNotFoundException fnfe = new FileNotFoundException("Archive cannot be found: "
77  					+ filename);
78  			fnfe.initCause(e);
79  			throw fnfe;
80  		}
81  
82  		if (filename.toLowerCase().endsWith(".gz")) {
83  			input = new GZIPInputStream(input);
84  		}
85  
86  		in = new BufferedInputStream(input, buffer);
87  		in.mark(buffer);
88  		
89  		scanner = null;
90  	}

100 	private void getScanner() throws IOException {
101 		if (scanner == null) {
102 			if (in == null) {
103 				throw new IOException();
104 			}
105 			scanner = new Scanner(in, dsc);
106 		}

111 		getScanner();

115 	@Override
116 	public final boolean execute(OperandStack os) {
117 		if (in == null) {
118 			return true;
119 		}
120 		try {
121 			getScanner();
122 		} catch (IOException e) {
123 			error(os, new IOError(e));
124 			return true;
125 		}
126 		return Dispatcher.dispatch(os, this);
127 	}

147 	@Override
148 	public final String readLine() throws IOException {
149 		try {
150 			if (in == null) {
151 				return null;
152 			}

154 			if (reader == null) {

169 	@Override
170 	public final void flush() throws IOException {
171 		if (in != null) {
172 			int b;
173 			do {
174 				b = in.read();
175 			} while (b != -1);
176 
177 			if (filter) {
178 				close();
179 			}
180 		}
181 	}

183 	@Override
184 	public final void reset() throws IOException {
185 		try {
186 			if (in.markSupported()) {
187 				in.reset();
188 				return;
189 			}
190 		} catch (IOException e) {
191 			// continue to re-open file
192 		}
193 
194 		// re-open file
195 		try {
196 			if (in != null) {
197 				in.close();
198 				in = null;
199 			}
200 		} catch (IOException e) {
201 			// ignore
202 		}
203 
204 		init();
205 	}
