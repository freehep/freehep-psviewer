14  public class PSName extends PSSimple {
15  	private String value;
16  
17  	public PSName(String name) {
18  		this(name, false);
19  	}
20  
21  	public PSName(String name, boolean literal) {
22  		super("name", literal);
23  		value = name;
24  	}
25  
26  	@Override
27  	public boolean execute(OperandStack os) {
28  		if (isLiteral()) {
29  			os.push(this);
30  			return true;
31  		}
32  
33  		// ignore names starting with ESC
34  		if (value.charAt(0) == 0x1B) {
35  			return true;
36  		}
37  
38  		PSObject obj = os.dictStack().lookup(this);
39  		if (obj == null) {
40  			error(os, new Undefined());
41  		}
42  
43  		os.execStack().pop();
44  		os.execStack().push(obj);
45  		return false;
46  	}
47  
48  	public int size() {
49  		return value.length();
50  	}
51  
52  	@Override
53  	public String getType() {
54  		return "nametype";
55  	}
56  
57  	public String getValue() {
58  		return value;
59  	}
60  
61  	@Override
62  	public Object clone() throws CloneNotSupportedException {
63  		return new PSName(value);
64  	}
65  
66  	@Override
67  	public boolean equals(Object o) {
68  		if (o instanceof PSName) {
69  			return getValue().equals(((PSName) o).getValue());
70  		} else if (o instanceof PSString) {
71  			return getValue().equals(((PSString) o).getValue());
72  		}
73  		return false;
74  	}
75  
76  	@Override
77  	public int hashCode() {
78  		return value.hashCode();
79  	}
80  
81  	@Override
82  	public String cvs() {
83  		return value;
84  	}
85  
86  	@Override
87  	public String toString() {
88  		return ((isLiteral()) ? "/" : "") + value;
89  	}
