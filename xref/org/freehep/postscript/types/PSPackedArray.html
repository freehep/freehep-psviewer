77  	private int execIndex = -1;

79  	@Override
80  	public boolean execute(OperandStack os) {
81  		if (isLiteral()) {
82  			os.push(this);
83  			return true;
84  		}
85  
86  		if (execIndex == -1) {
87  			// replace yourself with a copy, since the execIndex is a state
88  			// variable
89  			PSPackedArray copy;
90  			try {
91  				copy = (PSPackedArray) clone();
92  			} catch (CloneNotSupportedException e) {
93  				error(os, new Unimplemented());
94  				return true;
95  			}
96  			copy.setExecutable();
97  			copy.execIndex = 0;
98  			os.execStack().pop();
99  			os.execStack().push(copy);
100 			return false;
101 		}
102 
103 		if (execIndex < size()) {
104 			PSObject obj = get(execIndex);
105 			if ((obj instanceof PSPackedArray)
106 					&& ((PSPackedArray) obj).isExecutable()) {
107 				// subarrays (exec) are just pushed on the stack.
108 				os.push(obj);
109 			} else {
110 				os.execStack().push(obj);
111 			}
112 			execIndex++;
113 			return false;
114 		} else {
115 			execIndex = -1;
116 			return true;
117 		}
118 	}
