9   /**
10   * Objects for PostScript Processor, as defined in 3.3 Data Types and Objects
11   * 
12   * @author Mark Donszelmann

14  public class PSRandomAccessFile extends PSFile {
15  	private RandomAccessFile raf = null;
16  	private boolean write;
17  	private boolean append;
18  
19  	protected PSRandomAccessFile(String n, boolean f, RandomAccessFile r) {
20  		super(n, f, UNLIMITED);
21  		raf = r;
22  	}
23  
24  	public PSRandomAccessFile(String filename, boolean write, boolean append,
25  			boolean secure) throws IOException {
26  		super(filename, false, UNLIMITED);
27  
28  		if (!secure) {
29  			throw new IOException();
30  		}
31  		this.write = write;
32  		this.append = append;
33  		File file = new File(filename);
34  		if (!write && !file.exists()) {
35  			throw new IOException();
36  		}
37  		raf = new RandomAccessFile(file, "rw");
38  		if (append) {
39  			raf.seek(raf.length());
40  		}
41  		if (write) {
42  			// FIXME: not really truncated!
43  			raf.seek(0);
44  		}
45  	}
46  
47  	@Override
48  	public final void close() throws IOException {
49  		if (raf != null) {
50  			raf.close();
51  			raf = null;
52  		}
53  	}
54  
55  	@Override
56  	public final int read() throws IOException {
57  		return (raf != null) ? raf.read() : -1;
58  	}
59  
60  	@Override
61  	public final String readLine() throws IOException {
62  		return (raf != null) ? raf.readLine() : null;
63  	}
64  
65  	@Override
66  	public final void write(int b, boolean secure) throws IOException {
67  		if (!secure) {
68  			throw new IOException();
69  		}
70  		if (raf != null) {
71  			raf.write(b);
72  		} else {
73  			throw new IOException();
74  		}
75  	}
76  
77  	@Override
78  	public final void seek(long pos) throws IOException {
79  		if (raf != null) {
80  			raf.seek(pos);
81  		} else {
82  			throw new IOException();
83  		}
84  	}
85  
86  	@Override
87  	public final long getFilePointer() throws IOException {
88  		return (raf != null) ? raf.getFilePointer() : -1;
89  	}
90  
91  	@Override
92  	public final int available() throws IOException {
93  		return (raf != null) ? (int) (raf.length() - raf.getFilePointer()) : -1;
94  	}
95  
96  	@Override
97  	public void flush() throws IOException {
98  		// ignored
99  	}
100 
101 	@Override
102 	public final void reset() throws IOException {
103 		if (raf != null) {
104 			raf.seek(0);
105 		} else {
106 			throw new IOException();
107 		}
108 	}
109 
110 	@Override
111 	public final boolean isValid() {
112 		return (raf != null);
113 	}
114 
115 	@Override
116 	public final int hashCode() {
117 		return raf.hashCode();
118 	}
119 
120 	@Override
121 	public final boolean equals(Object o) {
122 		if (o instanceof PSRandomAccessFile) {
123 			return (raf == ((PSRandomAccessFile) o).raf);
124 		}
125 		return false;
126 	}
127 
128 	@Override
129 	public final Object clone() throws CloneNotSupportedException {
130 		return new PSRandomAccessFile(filename, filter, raf);
131 	}
132 
133 	@Override
134 	public final PSObject copy() {
135 		if (filter) {
136 			throw new IllegalArgumentException("Filters cannot be copied");
137 		}
138 
139 		try {
140 			return new PSRandomAccessFile(filename, write, append, true);
141 		} catch (FileNotFoundException e) {
142 			throw new IllegalArgumentException(
143 					"Cannot find file while copying: " + filename, e);
144 		} catch (IOException e) {
145 			throw new IllegalArgumentException(
146 					"IOException for file while copying: " + filename, e);
147 		}
148 	}
