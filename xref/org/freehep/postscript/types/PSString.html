232 	class ByteInputStream extends PushbackInputStream {
233 		private int initStart;
234 		private int initLength;
235 
236 		public ByteInputStream() {
237 			super(null, 1);
238 			initStart = start;
239 			initLength = length;
240 		}
241 
242 		// make bytes available one at a time
243 		@Override
244 		public int available() throws IOException {
245 			return (length > 0) ? 1 : 0;
246 		}
247 
248 		@Override
249 		public void close() throws IOException {
250 		}
251 
252 		@Override
253 		public synchronized void mark(int readLimit) {
254 		}
255 
256 		@Override
257 		public boolean markSupported() {
258 			return false;
259 		}
260 
261 		@Override
262 		public int read() throws IOException {
263 			if (length > 0) {
264 				int b = value[start] & 0xFF;
265 				start++;
266 				length--;
267 				return b;
268 			}
269 			return -1;
270 		}
271 
272 		@Override
273 		public int read(byte[] b) throws IOException {
274 			return read(b, 0, b.length);
275 		}
276 
277 		@Override
278 		public int read(byte[] b, int off, int len) throws IOException {
279 			if (b == null) {
280 				throw new IllegalArgumentException();
281 			} else if ((off < 0) || (off > b.length) || (len < 0)
282 					|| ((off + len) > b.length) || ((off + len) < 0)) {
283 				throw new IndexOutOfBoundsException();
284 			} else if (len == 0) {
285 				return 0;
286 			}
287 
288 			int c = read();
289 			if (c == -1) {
290 				return -1;
291 			}
292 			b[off] = (byte) c;
293 
294 			return 1;
295 		}
296 
297 		@Override
298 		public synchronized void reset() throws IOException {
299 			start = initStart;
300 			length = initLength;
301 		}
302 
303 		@Override
304 		public long skip(long n) throws IOException {
305 			if (n <= length) {
306 				start += n;
307 				length -= n;
308 				return n;
309 			}
310 			return -1;
311 		}
312 
313 		@Override
314 		public void unread(int b) throws IOException {
315 			if (start > 0) {
316 				start--;
317 				length++;
318 			}
319 		}
320 
321 		@Override
322 		public void unread(byte[] b) throws IOException {
323 			unread(b, 0, b.length);
324 		}
325 
326 		@Override
327 		public void unread(byte[] b, int off, int len) throws IOException {
328 			if (len > start) {
329 				throw new IOException();
330 			}
331 			start -= len;
332 			length += len;
333 		}
334 
335 	}
