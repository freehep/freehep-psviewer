13  	public static PSNumber parseNumber(String name)
14  			throws NumberFormatException {
15  
16  		try {
17  			int hash = name.indexOf('#');
18  			if (hash > 0) {
19  				int radix = Integer.parseInt(name.substring(0, hash));
20  				return new PSInteger(Integer.parseInt(name.substring(hash + 1),
21  						radix));
22  			} else {
23  				return new PSInteger(Integer.parseInt(name));
24  			}
25  		} catch (NumberFormatException nfe) {
26  		}
27  
28  		double d = Double.parseDouble(name);
29  		return new PSReal(d);
30  	}
