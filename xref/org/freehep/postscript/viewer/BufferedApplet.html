11  /**
12   * This class extends JApplet by adding double buffering. This is intended to be
13   * used in situations in which redrawing the contents of the panel is expensive.
14   * 
15   * @author Mark Donszelmann
16   */
17  public abstract class BufferedApplet extends Applet implements java.io.Serializable {
18  
19  	private Graphics offScreenGraphics;
20  
21  	private Image offScreenImage;
22  
23  	private Dimension oldDimension = new Dimension();
24  
25  	private Dimension dim = new Dimension();
26  
27  	private boolean printing = false;
28  
29  	private boolean repaint = false;
30  
31  	public BufferedApplet() {
32  		super();
33  	}
34  
35  	/**
36  	 * Triggers a full "user" repaint. If the "system" wants to repaint it will
37  	 * call the paint(Graphics) method directly, rather than scheduling a
38  	 * paint(Graphics) through a repaint().
39  	 */
40  	@Override
41  	public void repaint() {
42  		super.repaint();
43  		repaint = true;
44  	}
45  
46  	/**
47  	 * Triggers a full repaint, since the component is not valid anymore (size
48  	 * change, iconized, ...)
49  	 */
50  	@Override
51  	public void invalidate() {
52  		super.invalidate();
53  		repaint = true;
54  	}
55  
56  	/**
57  	 * Returns if true if paintComponent(Graphics) should be called (was
58  	 * triggered by a repaint() or invalidate(), and resets the trigger.
59  	 */
60  	private synchronized boolean shouldRepaint() {
61  		boolean result = repaint;
62  		repaint = false;
63  		return result;
64  	}
65  
66  	/**
67  	 * Paint this panel by calling paintComponent(Graphics) if necessary and
68  	 * flushing the buffered image to the screen. This method also handles
69  	 * printing and exporting separately.
70  	 * 
71  	 * @param g
72  	 *            Graphics object
73  	 */
74  	@Override
75  	public void paint(Graphics g) {
76  		super.paint(g);
77  
78  		// do not paint if params are null
79  		if ((g == null) || (offScreenImage == null)) {
80  			return;
81  		}
82  
83  		// decide where we are painting
84  		if (g instanceof PrinterGraphics) {
85  			printing = true;
86  		}
87  
88  		if (!isDisplaying()) {
89  			repaintComponent(g);
90  			return;
91  		}
92  
93  		if (shouldRepaint()) {
94  			repaintComponent(offScreenGraphics);
95  		}
96  
97  		g.drawImage(offScreenImage, 0, 0, this);
98  	}
99  
100 	/**
101 	 * Really repaint the component
102 	 * 
103 	 * @param graphics
104 	 */
105 	public abstract void repaintComponent(Graphics graphics);
106 
107 	/**
108 	 * Returns a pointer to the graphics (Graphics) context of the buffer. The
109 	 * user is NOT allowed to call dispose() on this graphics object.
110 	 * <P>
111 	 * NOTE: this method used to be called getGraphics, however, since the JVM
112 	 * paint thread may call getGraphics from paintImmediately and fails to work
113 	 * with our VectorGraphics context (the gc is not longer attached to the
114 	 * image), we decided to rename the method.
115 	 */
116 	public Graphics getOffscreenGraphics() {
117 		return offScreenGraphics;
118 	}
119 
120 	/**
121 	 * Resize and move a component.
122 	 */
123 	@Override
124 	public void setBounds(int x, int y, int w, int h) {
125 		// Make sure that the parent's method is called first;
126 		// otherwise, the resize never happens and new images are NOT
127 		// made.
128 		super.setBounds(x, y, w, h);
129 
130 		// Make the backing image.
131 		makeImage();
132 	}
133 
134 	/**
135 	 * Returns true if the drawing is made for a PrinterGraphics context.
136 	 */
137 	public boolean isPrinting() {
138 		return printing;
139 	}
140 
141 	/**
142 	 * Returns true if the drawing is made for a Graphics context, the display.
143 	 * True if not Printing.
144 	 */
145 	public boolean isDisplaying() {
146 		return !isPrinting();
147 	}

149 	/**
150 	 * Make the buffered image for this panel. Check to see if the size has
151 	 * changed before doing anything.
152 	 */
153 	private void makeImage() {
154 
155 		// Get the full size of the panel.
156 		dim = getSize(dim);
157 //		System.err.println(dim);
158 		
159 		// Check that the current size is positive and that the new
160 		// dimension is not equal to the old one.
161 		if (dim.width > 0 && dim.height > 0) {
162 			if (!oldDimension.equals(dim)) {
163 
164 				// allocate an Image, which from before JDK 1.4 was always in
165 				// System Memory,
166 				// and from JDK 1.4 is a Managed Buffered Image in System memory
167 				// or VRAM (Windows)
168 				// or a BufferedImage (getClass always returns BufferedImage)
169 				// Real BufferedImages resides in System Memory.
170 				// Volatile Images reside in VRAM (Windows), or XServer memory.
171 				// Drawing more than ~1000 line segments to system memory (BI)
172 				// and
173 				// doing a copy to the display is faster than drawing them
174 				// to video ram (VI) and using the fast VRAM->display copy.
175 				// For us, System Memory is the choice for both Windows and
176 				// XWindows,
177 				// since we draw more lines that copying.
178 
179 				// However, it may be that the acceleration possible on VRAM my
180 				// up this
181 				// 1000 line limit.
182 
183 				// The image created here is a Managed Buffered Image, which can
184 				// be moved
185 				// from System Memory and copied to VRAM...
186 				// Use the flag -Dsun.java2d.ddoffscreen=false to keep the image
187 				// in
188 				// system memory, or -Dsun.java2d.ddforcevram=true to keep the
189 				// image
190 				// in VRAM. Using none, seems to affect performance in picking
191 				// mode.
192 
193 				// There is no programmatic (java) way of keeping the image in
194 				// system memory, not even if you create a BufferedImage
195 				// explicitly.
196 				if (isOpaque()) {
197 					offScreenImage = super.createImage(dim.width, dim.height);
198 				} else {
199 					offScreenImage = new BufferedImage(dim.width, dim.height,
200 							BufferedImage.TYPE_INT_ARGB);
201 				}
202 				offScreenGraphics = offScreenImage.getGraphics();
203 
204 				// Reset the old size of this panel.
205 				oldDimension.setSize(dim);
206 			}
207 		} else {
208 			offScreenImage = null;
209 			offScreenGraphics = null;
210 		}
211 	}
