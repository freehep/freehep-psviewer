21  /**
22   * Texture Paint for PostScript Processor
23   * 
24   * @author Mark Donszelmann
25   */
26  public class FixedTexturePaint extends TexturePaint {
27  	private Logger log = Logger.getLogger("org.freehep.postscript");
28  
29  	private OperandStack os;
30  	private AffineTransform matrix;
31  	private Map<Color, TexturePaint> byColor = new HashMap<Color, TexturePaint>(
32  			2);
33  
34  	public FixedTexturePaint(OperandStack os, AffineTransform m,
35  			BufferedImage texture, Rectangle2D anchor) {
36  		super(texture, anchor);
37  		this.os = os;
38  		matrix = m;
39  	}
40  
41  	@Override
42  	public PaintContext createContext(ColorModel cm, Rectangle deviceBounds,
43  			Rectangle2D userBounds, AffineTransform xform, RenderingHints hints) {
44  		AffineTransform mirror;
45  		try {
46  			// calculate Mirror: x = M*ctm --> M = x*ctm-1
47  			mirror = os.gstate().getTransform();
48  			mirror = mirror.createInverse();
49  			mirror.preConcatenate(xform);
50  		} catch (NoninvertibleTransformException e) {
51  			log.warning("Pattern problem: could not invert matrix");
52  			mirror = xform;
53  		}
54  		AffineTransform at = new AffineTransform();
55  		at.preConcatenate(matrix);
56  		at.preConcatenate(mirror);
57  		return super.createContext(cm, deviceBounds, userBounds, at, hints);
58  	}
59  
60  	public TexturePaint inColor(Color color) {
61  		TexturePaint paint = byColor.get(color);
62  		if (paint == null) {
63  			paint = new TexturePaint(changeColor(super.getImage(), 0xFF000000,
64  					color.getRGB()), getAnchorRect());
65  			byColor.put(color, paint);
66  		}
67  		return paint;
68  	}
69  
70  	private BufferedImage changeColor(BufferedImage src, int oldColor,
71  			int newColor) {
72  		int width = src.getWidth();
73  		int height = src.getHeight();
74  		int type = src.getType();
75  		WritableRaster srcRaster = src.getRaster();
76  
77  		ColorModel dstColorModel = src.getColorModel();
78  		BufferedImage dst = new BufferedImage(src.getWidth(), src.getHeight(),
79  				BufferedImage.TYPE_INT_ARGB);
80  		WritableRaster dstRaster = dst.getRaster();
81  
82  		// replace oldcolor with newcolor and make others transparent
83  		int[] inPixels = new int[width];
84  		for (int y = 0; y < height; y++) {
85  			srcRaster.getDataElements(0, y, width, 1, inPixels);
86  			for (int x = 0; x < width; x++) {
87  				inPixels[x] = inPixels[x] == oldColor ? newColor
88  						: inPixels[x] & 0x00FFFFFF;
89  			}
90  			dstRaster.setDataElements(0, y, width, 1, inPixels);
91  		}
92  
93  		return dst;
94  	}
