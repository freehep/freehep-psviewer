87  	/*
88  	 * (non-Javadoc)
89  	 * 
90  	 * @see java.applet.Applet#init()
91  	 */
92  	@Override
93  	public void init() {
94  		try {
95  			String version = PSViewer.getVersion();
96  			// Popup Menu
97  			final URL aboutURL = new URL(
98  					"http://freehep.github.com/freehep-psviewer");
99  			menu = new PopupMenu();
100 			add(menu);
101 
102 			// about
103 			MenuItem aboutMenu = new MenuItem(
104 					"About the FreeHEP Postscript Viewer " + version + "...");
105 			menu.add(aboutMenu);
106 			aboutMenu.addActionListener(new ActionListener() {
107 				public void actionPerformed(ActionEvent e) {
108 					getAppletContext().showDocument(aboutURL);
109 				}
110 			});
111 			enableEvents(AWTEvent.MOUSE_EVENT_MASK);
112 
113 			String file = getParameter(FILE, "examples/cookbook/program_00.ps");
114 			int pageNo = getParameter(PAGE, 1);
115 			double sx = getParameter(SX, 1.0);
116 			double sy = getParameter(SY, 1.0);
117 			double tx = getParameter(TX, 0.0);
118 			double ty = getParameter(TY, 0.0);
119 			int buffer = getParameter(BUFFER, 0x8000);
120 			final URL url = new URL(getDocumentBase(), file);
121 
122 			MenuItem urlMenu = new MenuItem(url.toExternalForm());
123 			menu.add(urlMenu);
124 			urlMenu.addActionListener(new ActionListener() {
125 				public void actionPerformed(ActionEvent e) {
126 					getAppletContext().showDocument(url);
127 				}
128 			});
129 
130 			Processor processor = new Processor(this);
131 			new PSViewer(processor, url, pageNo, sx, sy, tx, ty, buffer, false);
132 		} catch (IOException e) {
133 			showStatus(e.getMessage());
134 		}
135 		repaint();
136 	}

146 	private String getParameter(String name, String defaultParameter) {
147 		String p = getParameter(name);

149 	}

151 	private int getParameter(String name, int defaultParameter) {
152 		String p = getParameter(name);
153 		if (p == null) {
154 			return defaultParameter;
155 		}
156 		try {
157 			return Integer.parseInt(p);
158 		} catch (NumberFormatException e) {
159 			return defaultParameter;
160 		}
161 	}
162 
163 	private double getParameter(String name, double defaultParameter) {
164 		String p = getParameter(name);
165 		if (p == null) {
166 			return defaultParameter;
167 		}
168 		try {
169 			return Double.parseDouble(p);
170 		} catch (NumberFormatException e) {
171 			return defaultParameter;
172 		}
