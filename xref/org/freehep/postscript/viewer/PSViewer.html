34  	public PSViewer(Processor processor, String name, int pageNo, double sx,
35  			double sy, double tx, double ty, int buffer, boolean debug) throws IOException {
36  		this(processor, new PSInputFile(name, buffer), pageNo, sx, sy, tx, ty, debug);
37  	}

39  	public PSViewer(Processor processor, URL url, int pageNo, double sx,
40  			double sy, double tx, double ty, int buffer, boolean debug) throws IOException {
41  		this(processor, new PSInputFile(url.toExternalForm(), buffer), pageNo, sx, sy,
42  				tx, ty, debug);
43  	}
44  
45  	public PSViewer(Processor processor, PSInputFile data, int pageNo,
46  			double sx, double sy, double tx, double ty, boolean debug)
47  			throws IOException {
48  		processor.setData(data);
49  		processor.setPageNo(pageNo);
50  		processor.setScale(sx, sy);
51  		processor.setTranslation(tx, ty);
52  		if (debug) {
53  			PSDebugger debugger = new PSDebugger();
54  			processor.attach(debugger);
55  			debugger.showInFrame();
56  			processor.reset();
57  		} else {
58  			processor.process();
59  		}
60  	}

62  	public static void main(String args[]) throws Exception {
63  		BooleanOption help = new BooleanOption("-help", "-h",
64  				"Show this help page", true);
65  		BooleanOption version = new BooleanOption("-version", "-v",
66  				"Show product version", true);
67  		BooleanOption debug = new BooleanOption("-debug", "-d",
68  				"Run with debugger");
69  		DoubleOption scale = new DoubleOption("-scale", "-s", "factor", 1.0,
70  				"Scale output by factor");
71  		DoubleOption tx = new DoubleOption("-transX", "-tx", "dx", 0.0,
72  				"Displace output in x by dx");
73  		DoubleOption ty = new DoubleOption("-transY", "-ty", "dy", 0.0,
74  				"Displace output in y by dy");
75  		IntOption page = new IntOption("-page", "-p", "#", 1,
76  				"Page to be displayed");
77  		IntOption buffer = new IntOption("-buffer", "-b", 0x8000,
78  				"Buffer size to use for document");
79  		StringParameter file = new StringParameter("file", "PostScript File");
80  
81  		ArgumentParser parser = new ArgumentParser(
82  				"org.freehep.postscript.PSViewer");
83  		parser.add(help);
84  		parser.add(version);
85  		parser.add(debug);
86  		parser.add(scale);
87  		parser.add(tx);
88  		parser.add(ty);
89  		parser.add(page);
90  		parser.add(buffer);
91  		parser.add(file);
92  
93  		try {
94  			List<?> extra = parser.parse(args);
95  
96  			if (!extra.isEmpty() || help.getValue()) {
97  				parser.printUsage(System.out);
98  				return;
99  			}
100 
101 			if (version.getValue()) {
102 				log.info("FreeHEP PSViewer "+getVersion());
103 				return;
104 			}
105 		} catch (MissingArgumentException mae) {
106 			log.warning(mae.getMessage());
107 			return;
108 		} catch (ArgumentFormatException afe) {
109 			log.warning(afe.getMessage());
110 			return;
111 		}
112 
113 		String name = file.getValue();
114 
115 		try {
116 			PSPanel panel = new PSPanel();
117 			panel.setBackground(Color.WHITE);
118 			JFrame frame = new JFrame("PostScript Output");
119 			frame.getContentPane().add(panel);
120 			frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
121 			frame.setSize(800, 700);
122 			frame.setVisible(true);
123 
124 			Processor processor = new Processor(panel);
125 			new PSViewer(processor, name, page.getValue().intValue(), scale
126 					.getValue().doubleValue(), scale.getValue().doubleValue(),
127 					tx.getValue().doubleValue(), ty.getValue().doubleValue(), buffer.getValue().intValue(),
128 					debug.getValue());
129 			panel.repaint();
130 		} catch (FileNotFoundException fnfe) {
131 			log.warning("File: '" + name + "' cannot be found.");
132 			System.exit(1);
133 		}
134 	}

141 		String version = "";
142 		InputStream is = PSViewer.class.getResourceAsStream("/META-INF/maven/org.freehep/freehep-psviewer/pom.properties");
143 		if (is != null) {
144 			Properties pom = new Properties();
145 			pom.load(is);
146 			is.close();
147 			version = pom.getProperty("version", "");
148 		}
149 		return version;
